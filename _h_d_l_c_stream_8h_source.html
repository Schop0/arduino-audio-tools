175 int charReceiver(uint8_t data) {
176 int result = 0;
177 uint8_t *frame_buffer_data = frame_buffer.address();
178 LOGD("charReceiver: %c", data);
179
180 if (data == FRAME_BOUNDARY_OCTET) {
181 if (escape_character == true) {
182 escape_character = false;
183 } else if ((frame_position >= 2) &&
184 (frame_checksum ==
185 ((frame_buffer_data[frame_position - 1] << 8) |
186 (frame_buffer_data[frame_position - 2] & 0xff)))) {
187 // trigger processing of result data
188 frame_buffer.setAvailable(frame_position - 2);
189 result = frame_buffer.available();
190 LOGD("==> Result: %d", result);
191 }
192 frame_position = 0;
193 frame_checksum = CRC16_CCITT_INIT_VAL;
194 return result;
195 }
196
197 if (escape_character) {
198 escape_character = false;
199 data ^= INVERT_OCTET;
200 } else if (data == CONTROL_ESCAPE_OCTET) {
201 escape_character = true;
202 return result;
203 }
204
205 frame_buffer_data[frame_position] = data;
206
207 if (frame_position - 2 >= 0) {
208 frame_checksum = _crc_ccitt_update(frame_checksum,
209 frame_buffer_data[frame_position - 2]);
210 }
211
212 frame_position++;
213
214 if (frame_position == max_frame_length) {
215 frame_position = 0;
216 frame_checksum = CRC16_CCITT_INIT_VAL;
217 }
218 return result;
219 }

222 void sendFrame(const uint8_t *framebuffer, size_t frame_length) {
223 LOGD("HDLCStream::sendFrame: %zu", frame_length);
224 uint8_t data;
225 uint16_t fcs = CRC16_CCITT_INIT_VAL;
226
227 p_out->write((uint8_t)FRAME_BOUNDARY_OCTET);
228
229 while (frame_length) {
230 data = *framebuffer++;
231 fcs = HDLCStream::_crc_ccitt_update(fcs, data);
232 if ((data == CONTROL_ESCAPE_OCTET) || (data == FRAME_BOUNDARY_OCTET)) {
233 p_out->write((uint8_t)CONTROL_ESCAPE_OCTET);
234 data ^= INVERT_OCTET;
235 }
236 p_out->write((uint8_t)data);
237 frame_length--;
238 }
239 data = low(fcs);
240 if ((data == CONTROL_ESCAPE_OCTET) || (data == FRAME_BOUNDARY_OCTET)) {
241 p_out->write((uint8_t)CONTROL_ESCAPE_OCTET);
242 data ^= (uint8_t)INVERT_OCTET;
243 }
244 p_out->write((uint8_t)data);
245 data = high(fcs);
246 if ((data == CONTROL_ESCAPE_OCTET) || (data == FRAME_BOUNDARY_OCTET)) {
247 p_out->write(CONTROL_ESCAPE_OCTET);
248 data ^= INVERT_OCTET;
249 }
250 p_out->write(data);
251 p_out->write(FRAME_BOUNDARY_OCTET);
252 p_out->flush();
253 }

255 static uint16_t crc16_update(uint16_t crc, uint8_t a) {
256 int i;
257 crc ^= a;
258 for (i = 0; i < 8; ++i) {
259 if (crc & 1)
260 crc = (crc >> 1) ^ 0xA001;
261 else
262 crc = (crc >> 1);
263 }
264 return crc;
265 }

267 static uint16_t crc_xmodem_update(uint16_t crc, uint8_t data) {
268 int i;
269
270 crc = crc ^ ((uint16_t)data << 8);
271 for (i = 0; i < 8; i++) {
272 if (crc & 0x8000)
273 crc = (crc << 1) ^ 0x1021;
274 else
275 crc <<= 1;
276 }
277
278 return crc;
279 }

281 static uint16_t _crc_ccitt_update(uint16_t crc, uint8_t data) {
282 data ^= lo8(crc);
283 data ^= data << 4;
284 return ((((uint16_t)data << 8) | hi8(crc)) ^ (uint8_t)(data >> 4) ^
285 ((uint16_t)data << 3));
286 }
