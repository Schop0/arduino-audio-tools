165 int parseString(const uint8_t* str, int len, int fileOffset = 0,
166 int level = 0) {
167 char type[5];
168 int idx = 0;
169 Box box;
170 while (true) {
171 if (!isValidType((const char*)str + idx + 4)) {
172 return idx;
173 }
174 size_t box_size = readU32(str + idx) - 8;
175 box.data = str + 8 + idx;
176 box.size = box_size;
177 box.level = level;
178 box.data_size = box.size;
179 box.file_offset = fileOffset + idx;
180 strncpy(box.type, (char*)(str + idx + 4), 4);
181 box.type[4] = '\0';
182 idx += box.size;
183 processCallback(box);
184 if (idx >= len) break; // No more data to parse
185 }
186 return idx;
187 }

190 bool findBox(const char* name, const uint8_t* data, size_t len, Box& result) {
191 for (int j = 0; j < len - 4; j++) {
192 if (!isValidType((const char*)data + j + 4)) {
193 continue; // Skip invalid types
194 }
195 size_t box_size = readU32(data + j) - 8;
196 if (box_size < 8) continue; // Invalid box size
197 Box box;
198 box.data = data + j + 8;
199 box.size = box_size;
200 box.data_size = box.size;
201 strncpy(box.type, (char*)(data + j + 4), 4);
202 box.type[4] = '\0';
203 if (StrView(box.type) == name) {
204 result = box;
205 return true; // Found the box
206 }
207 }
208 return false;
209 }

263 static uint32_t readU32(const uint8_t* p) {

279 virtual void parse() {
280 while (true) {
281 size_t bufferSize = buffer.available();
282 // not enough data to parse a box header
283 if (parseOffset + 8 > bufferSize) break;
284
285 parseOffset = checkParseOffset();
286
287 const uint8_t* p = buffer.data() + parseOffset;
288 uint32_t size32 = readU32(p);
289 char type[5];
290 strncpy(type, (char*)(p + 4), 4);
291
292 uint64_t boxSize = size32;
293 size_t headerSize = 8;
294 bool is_container = isContainerBox(type);
295
296 // check data in buffer
297 if (!is_container) {
298 // not enough data to provide data
299 if (boxSize > buffer.size()) {
300 LOGE("MP4Parser: '%s' box size %u exceeds buffer size %u", type,
301 (unsigned)boxSize, (unsigned)buffer.size());
302 is_error = true;
303 return;
304 }
305 // not enough data in buffer
306 if (parseOffset + boxSize > bufferSize) break;
307 }
308
309 // fill box with data
310 // Calculate absolute file offset for this box
311 uint64_t absBoxOffset = fileOffset + parseOffset;
312 int level = levelStack.size();
313
314 strcpy(box.type, type);
315 box.id++;
316 box.data = p + headerSize;
317 box.size = static_cast<size_t>(boxSize - headerSize);
318 box.data_size = box.size;
319 box.level = level;
320 box.file_offset = fileOffset + parseOffset;
321 box.is_complete = (parseOffset + boxSize <= bufferSize);
322 box.is_container = is_container;
323
324 // Special logic for container: usually no data
325 if (box.is_container) {
326 box.data_size = getContainerDataLength(box.type);
327 if (box.data_size == 0) box.data = nullptr;
328 box.is_complete = true;
329 }
330
331 // Regular logic for box with complete data
332 processCallback(box);
333
334 // Recurse into container
335 if (box.is_container) {
336 levelStack.push_back(absBoxOffset + boxSize);
337 parseOffset += (headerSize + box.data_size);
338 continue;
339 }
340
341 parseOffset += boxSize;
342
343 popLevels();
344 }
345
346 if (parseOffset > 0) {
347 fileOffset += parseOffset;
348 buffer.clearArray(parseOffset);
349 parseOffset = 0;
350 }
351 }

356 void popLevels() {
357 // Pop levels if we've passed their bounds (absolute file offset)
358 while (!levelStack.empty() &&
359 (fileOffset + parseOffset) >= levelStack.back()) {
360 levelStack.pop_back();
361 }
362 }

370 void processCallback(Box& box) {
371 bool is_called = false;
372 bool call_generic = true;
373 for (const auto& entry : callbacks) {
374 if (strncmp(entry.type, box.type, 4) == 0) {
375 entry.cb(box, ref);
376 is_called = true;
377 if (!entry.callGeneric) call_generic = false;
378 }
379 }
381 if ((!is_called || call_generic) && callback) callback(box, ref);
382 }

389 bool isContainerBox(const char* type) {
390 // fill with default values if nothing has been defined
391 if (containers.empty()) {
392 // pure containers
393 static const char* containers_str[] = {
394 "moov", "trak", "mdia", "minf", "stbl", "edts", "dinf", "udta",
395 "ilst", "moof", "traf", "mfra", "tref", "iprp", "sinf", "schi"};
396 for (const char* c : containers_str) {
397 ContainerInfo info;
398 info.name = c;
399 info.start = 0;
400 containers.push_back(info);
401 }
402 // container with data
403 ContainerInfo info;
404 info.name = "meta";
405 info.start = 4; // 4 bytes: version (1 byte) + flags (3 bytes)
406 containers.push_back(info);
407 }
408 // find the container by name
409 for (auto& cont : containers) {
410 if (StrView(type) == cont.name) return true;
411 }
412 return false;

444 size_t checkParseOffset() {
445 size_t current = parseOffset;
446 const char* type = (char*)(buffer.data() + parseOffset + 4);
447 for (int j = 0; j < buffer.available() - parseOffset - 4; j += 4) {
448 if (isValidType(type, j)) {
449 if (j != 0) {
450 // report the data under the last valid box
451 box.size = 0;
452 box.data_size = j;
453 box.level = static_cast<int>(levelStack.size()) + 1;
454 box.data = buffer.data() + parseOffset;
455 processCallback(box);
456 }
457
458 return j + parseOffset;
459 }
460 }
461 return parseOffset;
