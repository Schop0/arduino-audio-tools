309 int charReceiver(uint8_t data) {
310 int result = 0;
311 uint8_t *frame_buffer_data = frame_buffer.address();
312 LOGD("charReceiver: %c", data);
313
314 if (data == FRAME_BOUNDARY_OCTET) {
315 if (escape_character == true) {
316 escape_character = false;
317 } else if ((frame_position >= 2) &&
318 (frame_checksum ==
319 ((frame_buffer_data[frame_position - 1] << 8) |
320 (frame_buffer_data[frame_position - 2] & 0xff)))) {
321 // trigger processing of result data
322 frame_buffer.setAvailable(frame_position - 2);
323 result = frame_buffer.available();
324 LOGD("==> Result: %d", result);
325 }
326 frame_position = 0;
327 frame_checksum = CRC16_CCITT_INIT_VAL;
328 return result;
329 }
330
331 if (escape_character) {
332 escape_character = false;
333 data ^= INVERT_OCTET;
334 } else if (data == CONTROL_ESCAPE_OCTET) {
335 escape_character = true;
336 return result;
337 }
338
339 frame_buffer_data[frame_position] = data;
340
341 if (frame_position - 2 >= 0) {
342 frame_checksum = crc_ccitt_update(frame_checksum,
343 frame_buffer_data[frame_position - 2]);
344 }
345
346 frame_position++;
347
348 if (frame_position == max_frame_length) {
349 LOGE("buffer overflow: %d", frame_position);
350 frame_position = 0;
351 }
352 return result;

356 void sendFrame(const uint8_t *framebuffer, size_t frame_length) {
357 LOGD("HDLCStream::sendFrame: %zu", frame_length);
358 uint8_t data;
359 uint16_t fcs = CRC16_CCITT_INIT_VAL;
360
361 p_out->write((uint8_t)FRAME_BOUNDARY_OCTET);
362
363 while (frame_length) {
364 data = *framebuffer++;
365 fcs = HDLCStream::crc_ccitt_update(fcs, data);
366 if ((data == CONTROL_ESCAPE_OCTET) || (data == FRAME_BOUNDARY_OCTET)) {
367 p_out->write((uint8_t)CONTROL_ESCAPE_OCTET);
368 data ^= INVERT_OCTET;
369 }
370 p_out->write((uint8_t)data);
371 frame_length--;
372 }
373 data = low(fcs);
374 if ((data == CONTROL_ESCAPE_OCTET) || (data == FRAME_BOUNDARY_OCTET)) {
375 p_out->write((uint8_t)CONTROL_ESCAPE_OCTET);
376 data ^= (uint8_t)INVERT_OCTET;
377 }
378 p_out->write((uint8_t)data);
379 data = high(fcs);
380 if ((data == CONTROL_ESCAPE_OCTET) || (data == FRAME_BOUNDARY_OCTET)) {
381 p_out->write(CONTROL_ESCAPE_OCTET);
382 data ^= INVERT_OCTET;
383 }
384 p_out->write(data);
385 p_out->write(FRAME_BOUNDARY_OCTET);
386 p_out->flush();
387 }

398 static uint16_t crc_ccitt_update(uint16_t crc, uint8_t data) {
399 data ^= lo8(crc);
400 data ^= data << 4;
401 return ((((uint16_t)data << 8) | hi8(crc)) ^ (uint8_t)(data >> 4) ^
402 ((uint16_t)data << 3));
