240 void processFtyp(uint8_t byte) {
241 static uint8_t ftyp_signature[] = {'f', 't', 'y', 'p'};
242 static int signature_pos = 0;
243
244 // Skip first 4 bytes (box size)
245 static int header_pos = 0;
246 if (header_pos < 4) {
247 header_pos++;
248 file_offset++;
249 return;
250 }
251
252 // Check for ftyp signature
253 if (byte == ftyp_signature[signature_pos]) {
254 signature_pos++;
255 file_offset++;
256 if (signature_pos == 4) {
257 // Found ftyp, move to general box parsing
258 resetBoxHeader();
259 state = ParserState::READING_BOX_HEADER;
260 signature_pos = 0;
261 header_pos = 0;
262 }
263 } else {
264 // Reset if signature doesn't match
265 signature_pos = 0;
266 file_offset++;
267 }
268 }

277 void processBoxHeader(uint8_t byte) {
278 static int header_pos = 0;
279
280 if (header_pos < 4) {
281 // Reading size (big endian)
282 current_box.size = (current_box.size << 8) | byte;
283 } else if (header_pos < 8) {
284 // Reading box type (4 characters)
285 current_box.type[header_pos - 4] = byte;
286 }
287
288 header_pos++;
289 file_offset++;
290
291 if (header_pos == 8) {
292 // Header complete
293 header_pos = 0;
294 LOGI("Found box: %s, size: %u, offset: %llu", current_box.type,
295 current_box.size, current_box.absolute_offset);
296
297 // Handle special box sizes
298 if (current_box.size == 1) {
299 // Extended size in next 8 bytes (64-bit) - not handling for simplicity
300 LOGW("64-bit box size not fully supported");
301 } else if (current_box.size == 0) {
302 // Box extends to end of file - not handling for simplicity
303 LOGW("Box extends to end of file - not fully supported");
304 }
305
306 // Process based on box type
307 if (strcmp(current_box.type, BOX_MDAT) == 0) {
308 mdat_start_offset = file_offset;
309 state = ParserState::READING_MDAT;
310 current_box.bytes_read = 8; // Account for the header we just read
311 } else {
312 // Check if this is a container box
313 bool is_container = isContainerBox(current_box.type);
314
315 if (is_container) {
316 // Push current box to stack and continue reading boxes inside it
317 box_stack.push_back(current_box);
318 state = ParserState::READING_BOX_HEADER;
319 resetBoxHeader();
320 } else {
321 // Standard data box
322 state = ParserState::READING_BOX_DATA;
323 buffer_pos = 0;
324 current_box.bytes_read = 8; // Account for the header we just read
325 }
326 }
327 }

337 void processBoxData(uint8_t byte) {
338 // Add byte to buffer if space available
339 if (buffer_pos < buffer_size) {
340 buffer[buffer_pos++] = byte;
341 }
342
343 current_box.bytes_read++;
344 file_offset++;
345
346 // If we've read the entire box
347 if (current_box.bytes_read >= current_box.size) {
348 // Process the completed box
349 processCompletedBox();
350
351 // Check if we need to pop from box stack
352 if (!box_stack.empty()) {
353 BoxHeader& parent = box_stack.back();
354 parent.bytes_read += current_box.size;
355
356 // If parent is complete, process it
357 if (parent.bytes_read >= parent.size) {
358 processCompletedParentBox(parent);
359 box_stack.pop_back();
360
361 // Keep popping if higher level boxes are also complete
362 while (!box_stack.empty()) {
363 BoxHeader& grandparent = box_stack.back();
364 grandparent.bytes_read += parent.size;
365
366 if (grandparent.bytes_read >= grandparent.size) {
367 processCompletedParentBox(grandparent);
368 box_stack.pop_back();
369 parent = grandparent;
370 } else {
371 break;
372 }
373 }
374 }
375 }
376
377 // Move back to reading box headers
378 state = ParserState::READING_BOX_HEADER;
379 resetBoxHeader();
380 }
381 }

413 // Parse AudioSpecificConfig from esds box
414 void parseEsdsBox() {
415 if (buffer_pos < 31) {
416 LOGW("ESDS box too small");
417 return;
418 }
419
420 // The structure of the ESDS box is complex with nested descriptors
421 // For simplicity, we assume a fixed structure to find the
422 // AudioSpecificConfig
423
424 // Search for the config
425 for (size_t i = 0; i < buffer_pos - 2; i++) {
426 // Look for the audio config pattern: typical descriptor structure
427 if (buffer[i] == 0x05 && buffer[i + 1] == 0x02) {
428 // Found AudioSpecificConfig (ESDescriptor with tag 0x05)
429 audio_config[0] = buffer[i + 2];
430 audio_config[1] = buffer[i + 3];
431 has_audio_config = true;
432 result_codec_type = M4ACodecType::AAC;
433
434 LOGI("Found AAC config: %02X %02X", audio_config[0], audio_config[1]);
435 break;
436 }
437 }
438 }

440 // Parse ALAC configuration
441 void parseAlacBox() {
442 if (buffer_pos < 36) {
443 LOGW("ALAC box too small");
444 return;
445 }
446
447 // Store the ALAC configuration data
448 // Skip first 8 bytes (version and flags)
449 alac_config_size = min(buffer_pos - 8, sizeof(alac_config));
450 memcpy(alac_config, buffer.data() + 8, alac_config_size);
451 is_alac = true;
452 result_codec_type = M4ACodecType::ALAC;
453
454 LOGI("Found ALAC config, size: %d", alac_config_size);
455
456 // Extract basic audio parameters from config
457 uint32_t frameLength = readUint32(alac_config, 0);
458 uint8_t bitDepth = alac_config[9];
459 uint8_t channels = alac_config[13];
460 uint32_t sampleRate = readUint32(alac_config, 20);
461
462 // Update audio info
463 info.bits_per_sample = bitDepth;
464 info.channels = channels;
465 info.sample_rate = sampleRate;
466
467 LOGI("ALAC: %d Hz, %d bit, %d ch", info.sample_rate, info.bits_per_sample,
468 info.channels);
469
470 // Notify decoder about the codec info if available
471 if (p_decoder){
472 if (p_decoder->selectDecoder(mime())){
473 p_decoder->writeCodecConfig(alac_config, alac_config_size);
474 }
475 }

498 // Parse Sample Description Box to detect codec type
499 void parseStsdBox() {
500 if (buffer_pos < 16) {
501 LOGW("STSD box too small");
502 return;
503 }
504
505 uint32_t version_flags = readUint32(buffer.data(), 0);
506 uint32_t entry_count = readUint32(buffer.data(), 4);
507
508 if (entry_count == 0 || buffer_pos < 16 + 8) {
509 LOGW("No entries in STSD box");
510 return;
511 }
512
513 // Each entry starts with size (4 bytes) and type (4 bytes)
514 uint32_t entry_size = readUint32(buffer.data(), 8);
515 char entry_type[5] = {0};
516 memcpy(entry_type, buffer.data() + 12, 4);
517
518 LOGI("STSD entry: %s, size: %d", entry_type, entry_size);
519
520 // Check codec type
521 if (strcmp(entry_type, "mp4a") == 0) {
522 result_codec_type = M4ACodecType::AAC;
523 LOGI("Found AAC audio");
524 } else if (strcmp(entry_type, "alac") == 0) {
525 result_codec_type = M4ACodecType::ALAC;
526 LOGI("Found ALAC audio");
527 } else {
528 result_codec_type = M4ACodecType::UNKNOWN;
529 LOGW("Unknown audio codec: %s", entry_type);
530 }
531 }

689 void processMdat(uint8_t byte) {
690 // First check if we have valid sample tables and still have samples to
691 // process
692 if (current_sample >= sample_count || stsz_entries.empty() ||
693 stco_entries.empty()) {
694 // Skip MDAT if we don't have valid sample info
695 current_box.bytes_read++;
696 file_offset++;
697
698 if (current_box.bytes_read >= current_box.size) {
699 state = ParserState::READING_BOX_HEADER;
700 resetBoxHeader();
701 }
702 return;
703 }
704
705 // Check if we've reached the offset for the current sample
706 if (file_offset >= current_sample_offset) {
707 // Add byte to buffer
708 if (buffer_pos < buffer_size) {
709 buffer[buffer_pos++] = byte;
710 }
711
712 // Check if we've collected the entire sample
713 if (buffer_pos >= current_sample_size) {
714 // Process based on output mode and codec type
715 // Output compressed format
716 switch (result_codec_type) {
717 case M4ACodecType::AAC:
718 // Output AAC frame with ADTS header if needed
719 if (has_audio_config) {
720 if (!notified_config) {
721 notified_config = true;
722 LOGI("Outputting AAC with ADTS headers");
723 }
724 uint8_t adts_header[7];
725 createADTSHeader(adts_header, buffer_pos);
726 writeResult(adts_header, 7);
727 }
728 writeResult(buffer.data(), buffer_pos);
729 break;
730
731 case M4ACodecType::ALAC:
732 // Output ALAC frame
733 if (!notified_config) {
734 notified_config = true;
735 LOGI("Outputting raw ALAC frames");
736 }
737 writeResult(buffer.data(), buffer_pos);
738 break;
739
740 default:
741 // Unknown codec, output raw data
742 LOGW("Unknown codec type, outputting raw data");
743 writeResult(buffer.data(), buffer_pos);
744 break;
745 }
746 }
747
748 // Move to next sample
749 current_sample++;
750 buffer_pos = 0;
751
752 if (current_sample < sample_count) {
753 calculateNextSamplePosition();
754 }
755 }
756
757 // Increment counters
758 current_box.bytes_read++;
759 file_offset++;
760
761 // Check if we've read the entire MDAT box
762 if (current_box.bytes_read >= current_box.size) {
763 state = ParserState::READING_BOX_HEADER;
764 resetBoxHeader();
765 }
766 }

770 // if we have a decoder we only decode the supported type
771 const char* original_mime = p_decoder->selectedMime();
772 if (p_decoder != nullptr) {

774 LOGE("Unsupported mime type %s", mime());
775 return 0;

787 void calculateNextSamplePosition() {
788 // Get sample size
789 current_sample_size = stsz_entries[current_sample].sample_size;
790
791 // Find which chunk this sample belongs to
792 uint32_t chunk_idx = 0;
793 uint32_t samples_in_chunks_so_far = 0;
794 uint32_t samples_per_chunk = 0;
795
796 for (size_t i = 0; i < stsc_entries.size(); i++) {
797 StscEntry& entry = stsc_entries[i];
798 uint32_t next_first_chunk = (i < stsc_entries.size() - 1)
799 ? stsc_entries[i + 1].first_chunk
800 : stco_entries.size() + 1;
801
802 // Calculate number of chunks with this samples_per_chunk value
803 uint32_t chunk_count = next_first_chunk - entry.first_chunk;
804
805 // Check if our sample falls in this range
806 uint32_t samples_in_range = chunk_count * entry.samples_per_chunk;
807 if (current_sample < samples_in_chunks_so_far + samples_in_range) {
808 // Found the right entry
809 samples_per_chunk = entry.samples_per_chunk;
810
811 // Calculate which chunk within this range
812 uint32_t relative_chunk =
813 (current_sample - samples_in_chunks_so_far) / samples_per_chunk;
814 chunk_idx = entry.first_chunk - 1 +
815 relative_chunk; // Adjust for 1-based indexing
816 break;
817 }
818
819 samples_in_chunks_so_far += samples_in_range;
820 }
821
822 if (chunk_idx >= stco_entries.size()) {
823 LOGE("Invalid chunk index %d (max %d)", chunk_idx,
824 stco_entries.size() - 1);
825 return;
826 }
827
828 // Get chunk offset
829 uint64_t chunk_offset = stco_entries[chunk_idx].chunk_offset;
830
831 // Calculate sample offset within chunk
832 uint32_t sample_in_chunk =
833 (current_sample - samples_in_chunks_so_far) % samples_per_chunk;
834
835 // Sum sizes of preceding samples in this chunk
836 uint64_t offset_in_chunk = 0;
837 for (uint32_t i = 0; i < sample_in_chunk; i++) {
838 uint32_t preceding_sample = current_sample - sample_in_chunk + i;
839 offset_in_chunk += stsz_entries[preceding_sample].sample_size;
840 }
841
842 // Final sample offset
843 current_sample_offset = chunk_offset + offset_in_chunk;
844 }

846 void createADTSHeader(uint8_t* header, size_t aac_frame_size) {
847 // Extract values from AudioSpecificConfig
848 uint8_t profile = (audio_config[0] >> 3) & 0x1F; // 5 bits
849 uint8_t sampling_freq_index = ((audio_config[0] & 0x7) << 1) |
850 ((audio_config[1] >> 7) & 0x1); // 4 bits
851 uint8_t channel_config = (audio_config[1] >> 3) & 0xF; // 4 bits
852
853 // Adjust profile value for ADTS (profile in AudioSpecificConfig is
854 // profile-1)
855 profile = profile + 1;
856
857 // Create ADTS header (7 bytes)
858 header[0] = 0xFF; // Sync word
859 header[1] = 0xF1; // MPEG-4, Layer, Protection absent
860 header[2] = ((profile - 1) << 6) | (sampling_freq_index << 2) |
861 (channel_config >> 2);
862 header[3] = ((channel_config & 0x3) << 6) | ((aac_frame_size + 7) >> 11);
863 header[4] = ((aac_frame_size + 7) >> 3) & 0xFF;
864 header[5] = (((aac_frame_size + 7) & 0x7) << 5) | 0x1F;
865 header[6] = 0xFC;
866 }
