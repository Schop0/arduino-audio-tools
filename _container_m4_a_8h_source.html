131 static void decodeAudio(const M4AAudioDemuxer::Frame& frame, void* ref) {
132 ContainerM4A* self = static_cast<ContainerM4A*>(ref);
133 if (self->p_decoder == nullptr) {
134 self->p_print->write(frame.data, frame.size);
135 return;
136 }
137 MultiDecoder& dec = *(self->p_decoder);
138 const char* current_mime = dec.selectedMime();
139
140 // select decoder based on mime type
141 if (!dec.selectDecoder(frame.mime)) {
142 const char* mime = frame.mime ? frame.mime : "(nullptr)";
143 LOGE("No decoder found for mime type: %s", mime);
144 return;
145 }
146
147 // for ALAC only: process magic cookie if not done yet
148 if (StrView(frame.mime) == "audio/alac" &&
149 !self->is_magic_cookie_processed) {
150 auto& magic_cookie = self->demux.getALACMagicCookie();
151 if (magic_cookie.size() > 0) {
152 if (!dec.setCodecConfig(magic_cookie.data(), magic_cookie.size())) {
153 LOGE("Failed to set ALAC magic cookie for decoder: %s",
154 dec.selectedMime());
155 }
156 }
157 self->is_magic_cookie_processed = true;
158 }
159 // write encoded data to decoder
160 dec.write(frame.data, frame.size);
161
162 // restore the original mime type
163 dec.selectDecoder(current_mime);
164 }
