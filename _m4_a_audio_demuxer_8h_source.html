91 size_t write(const uint8_t* data, size_t len, bool is_final) {
92 // Resize buffer to the current sample size
93 size_t currentSize = currentSampleSize();
94 if (currentSize == 0) {
95 LOGE("No sample size defined, cannot write data");
96 return 0;
97 }
98 resize(currentSize);
99
101 for (int j = 0; j < len; j++) {
102 buffer.write(data[j]);
103 if (buffer.available() == currentSize) {
104 LOGI("Sample# %zu: size %zu bytes", sampleIndex, currentSize);
105 executeCallback(currentSize);
106 buffer.clear();
107 box_pos += currentSize;
108 ++sampleIndex;
109 currentSize = currentSampleSize();
110 if (box_pos >= box_size) {
111 LOGI("Reached end of box: %s write",
112 is_final ? "final" : "not final");
113 return j;
114 }
115 if (currentSize == 0) {
116 LOGE("No sample size defined, cannot write data");
117 return j;
118 }
119 resize(currentSize);
120 }
121 }
122 return len;
123 }

179 void executeCallback(size_t size) {
180 size_t frameSize = size;
181 Frame frame;
182 frame.codec = codec;
183 frame.data = buffer.data();
184 frame.size = frameSize;
185 frame.timestamp = 0; // TODO: timestamp
186 switch (codec) {
187 case Codec::AAC: {
188 uint8_t out[frameSize + 7];
189 writeAdtsHeader(out, aacProfile, sampleRateIdx, channelCfg,
190 frameSize);
191 memcpy(out + 7, buffer.data(), frameSize);
192 frame.data = out;
193 frame.size = sizeof(out);
194 frame.mime = "audio/aac";
195 if (callback)
196 callback(frame, ref);
197 else
198 LOGE("No callback defined for audio frame extraction");
199 return;
200 }
201 case Codec::ALAC:
202 frame.mime = "audio/alac";
203 break;
204 case Codec::MP3:
205 frame.mime = "audio/mpeg";
206 break;
207 default:
208 frame.mime = nullptr;
209 break;
210 }
211 if (callback)
212 callback(frame, ref);
213 else
214 LOGE("No callback defined for audio frame extraction");
215 }

251 static void writeAdtsHeader(uint8_t* adts, int aacProfile,
252 int sampleRateIdx, int channelCfg,
253 int frameLen) {
254 adts[0] = 0xFF;
255 adts[1] = 0xF1;
256 adts[2] = ((aacProfile - 1) << 6) | (sampleRateIdx << 2) |
257 ((channelCfg >> 2) & 0x1);
258 adts[3] = ((channelCfg & 0x3) << 6) | ((frameLen + 7) >> 11);
259 adts[4] = ((frameLen + 7) >> 3) & 0xFF;
260 adts[5] = (((frameLen + 7) & 0x7) << 5) | 0x1F;
261 adts[6] = 0xFC;
262 }

382 static void boxDataSetupCallback(MP4Parser::Box& box, void* ref) {
383 M4AAudioDemuxer& self = *static_cast<M4AAudioDemuxer*>(ref);
384
385 // mdat must not be buffered
386 if (StrView(box.type) == "mdat") {
387 LOGI("Box: %s, size: %u bytes", box.type, (unsigned)box.size);
388 self.sampleExtractor.setMaxSize(box.size);
389 return;
390 }
391
392 bool is_relevant = isRelevantBox(box.type);
393 if (is_relevant) {
394 LOGI("Box: %s, size: %u bytes", box.type, (unsigned)box.size);
395 if (box.data_size == 0) {
396 // setup for incremental processing
397 self.resize(box.size);
398 self.buffer.clear();
399 } else {
400 // we have the complete box data
401 self.processBox(box);
402 }
403 }
404 }
405
414 static void incrementalBoxDataCallback(MP4Parser::Box& box,
415 const uint8_t* data, size_t len,
416 bool is_final, void* ref) {
417 M4AAudioDemuxer& self = *static_cast<M4AAudioDemuxer*>(ref);
418
419 // mdat must not be buffered
420 if (StrView(box.type) == "mdat") {
421 LOGI("*Box: %s, size: %u bytes", box.type, (unsigned)len);
422 self.sampleExtractor.write(data, len, is_final);
423 return;
424 }
425
426 // only process relevant boxes
427 if (!isRelevantBox(box.type)) return;
428
429 LOGI("*Box: %s, size: %u bytes", box.type, (unsigned)len);
430
431 // others fill buffer incrementally
432 if (len > 0) {
433 size_t written = self.buffer.writeArray(data, len);
434 if (written != len) {
435 LOGE("Failed to write all data to buffer, written: %zu, expected: %zu",
436 written, len);
437 }
438 }
439
440 // on last chunk, call the specific box handler
441 if (is_final) {
442 MP4Parser::Box complete_box = box;
443 complete_box.data = self.buffer.data();
444 complete_box.data_size = self.buffer.size();
445 self.processBox(complete_box);
446 // The buffer might be quite large, so we resize it to the default size
447 self.resize(self.default_size);
448 }

470 void onStsd(const MP4Parser::Box& box) {
471 LOGI("onStsd: %s, size: %zu bytes", box.type, box.data_size);
472 const uint8_t* data = box.data;
473 size_t size = box.data_size;
474 if (size < 8) return;

484 void onMp4a(const MP4Parser::Box& box) {
485 LOGI("onMp4a: %s, size: %zu bytes", box.type, box.data_size);
486 if (box.data_size < 36) return; // Minimum size for mp4a box
487
488 // use default configuration
489 int aacProfile = 2; // Default: AAC LC
490 int sampleRateIdx = 4; // Default: 44100 Hz
491 int channelCfg = 2; // Default: Stereo
492 sampleExtractor.setAACConfig(aacProfile, sampleRateIdx, channelCfg);
493 codec = Codec::AAC;
494 sampleExtractor.setCodec(codec);
495
497 int pos = 36 - 8;
498 parser.parseString(box.data + pos, box.data_size - pos);
499 }

505 void onEsds(const MP4Parser::Box& box) {
506 LOGI("onEsds: %s, size: %zu bytes", box.type, box.data_size);
507 int aacProfile = 2; // Default: AAC LC
508 int sampleRateIdx = 4; // Default: 44100 Hz
509 int channelCfg = 2; // Default: Stereo
510
511 for (size_t i = 2; i + 4 < box.data_size; ++i) {
512 if (box.data[i] == 0x05) { // 0x05 = AudioSpecificConfig tag
513 uint8_t asc_len = box.data[i + 1];
514 if (i + 2 + asc_len > box.data_size) {
515 LOGW("esds box not long enough for AudioSpecificConfig");
516 break;
517 };
518 const uint8_t* asc = box.data + i + 2;
519 // AudioSpecificConfig is at least 2 bytes
520 aacProfile = (asc[0] >> 3) & 0x1F; // 5 bits
521 sampleRateIdx =
522 ((asc[0] & 0x07) << 1) | ((asc[1] >> 7) & 0x01); // 4 bits
523 channelCfg = (asc[1] >> 3) & 0x0F; // 4 bits
524 LOGI("AudioSpecificConfig: profile=%d, sampleRateIdx=%d, channelCfg=%d",
525 aacProfile, sampleRateIdx, channelCfg);
526 sampleExtractor.setAACConfig(aacProfile, sampleRateIdx, channelCfg);
527 }
528 }
529 }

548 void onStsz(MP4Parser::Box& box) {
549 LOGI("onStsz: %s, size: %zu bytes", box.type, box.data_size);
550 // Parse stsz box and fill sampleSizes
551 const uint8_t* data = box.data;
552 size_t size = box.data_size;
553 if (size < 12) return;
554 uint32_t sampleSize = readU32(data + 4);
555 uint32_t sampleCount = readU32(data + 8);
556 sampleExtractor.begin();
557 Vector<uint32_t>& sampleSizes = sampleExtractor.getSampleSizes();
558 if (sampleSize == 0) {
559 if (size < 12 + 4 * sampleCount) return;
560 LOGI("-> Sample Sizes Count: %u", sampleCount);
561 sampleSizes.resize(sampleCount);
562 for (uint32_t i = 0; i < sampleCount; ++i) {
563 sampleSizes[i] = readU32(data + 12 + i * 4);
564 }
565 } else {
566 sampleExtractor.setFixedSampleCount(sampleSize, sampleCount);
567 }
568 }

574 void onStco(MP4Parser::Box& box) {
575 LOGI("onStco: %s, size: %zu bytes", box.type, box.data_size);
576 // Parse stco box and fill chunkOffsets
577 const uint8_t* data = box.data + 4;
578 size_t size = box.data_size;
579 if (size < 4) return;
580 uint32_t entryCount = readU32(data);
581 Vector<uint32_t>& chunkOffsets = sampleExtractor.getChunkOffsets();
582 if (size < 4 + 4 * entryCount) return;
583 chunkOffsets.resize(entryCount);
584 LOGI("-> Chunk offsets count: %u", entryCount);
585 for (uint32_t i = 0; i < entryCount; ++i) {
586 chunkOffsets[i] = readU32(data + 4 + i * 4);
587 }
588 }
