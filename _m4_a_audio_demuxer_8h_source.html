47 // Parses esds content to extract audioObjectType, frequencyIndex, and
48 // channelConfiguration
49 bool parse(const uint8_t* data, size_t size) {
50 const uint8_t* ptr = data;
51 const uint8_t* end = data + size;
52
53 if (ptr + 4 > end) return false;
54 ptr += 4; // skip version + flags
55
56 if (ptr >= end || *ptr++ != 0x03) return false;
57 size_t es_len = parse_descriptor_length(ptr, end);
58 if (ptr + es_len > end) return false;
59
60 ptr += 2; // skip ES_ID
61 ptr += 1; // skip flags
62
63 if (ptr >= end || *ptr++ != 0x04) return false;
64 size_t dec_len = parse_descriptor_length(ptr, end);
65 if (ptr + dec_len > end) return false;
66
67 ptr += 13; // skip objectTypeIndication, streamType, bufferSizeDB,
68 // maxBitrate, avgBitrate
69
70 if (ptr >= end || *ptr++ != 0x05) return false;
71 size_t dsi_len = parse_descriptor_length(ptr, end);
72 if (ptr + dsi_len > end || dsi_len < 2) return false;
73
74 uint8_t byte1 = ptr[0];
75 uint8_t byte2 = ptr[1];
76
77 audioObjectType = (byte1 >> 3) & 0x1F;
78 samplingRateIndex = ((byte1 & 0x07) << 1) | ((byte2 >> 7) & 0x01);
79 channelConfiguration = (byte2 >> 3) & 0x0F;
80 return true;
81 }
82
83 protected:
84 // Helper to decode variable-length descriptor lengths (e.g. 0x80 80 80 05)
85 inline size_t parse_descriptor_length(const uint8_t*& ptr,
86 const uint8_t* end) {
87 size_t len = 0;
88 for (int i = 0; i < 4 && ptr < end; ++i) {
89 uint8_t b = *ptr++;
90 len = (len << 7) | (b & 0x7F);
91 if ((b & 0x80) == 0) break;
92 }
93 return len;
94 }
95 };

163 size_t write(const uint8_t* data, size_t len, bool is_final) {
164 // Resize buffer to the current sample size
165 size_t currentSize = currentSampleSize();
166 if (currentSize == 0) {
167 LOGE("No sample size defined: e.g. mdat before stsz!");
168 return 0;
169 }
170 resize(currentSize);
171
173 for (int j = 0; j < len; j++) {
174 buffer.write(data[j]);
175 if (buffer.available() == currentSize) {
176 LOGI("Sample# %zu: size %zu bytes", sampleIndex, currentSize);
177 executeCallback(currentSize);
178 buffer.clear();
179 box_pos += currentSize;
180 ++sampleIndex;
181 currentSize = currentSampleSize();
182 if (box_pos >= box_size) {
183 LOGI("Reached end of box: %s write",
184 is_final ? "final" : "not final");
185 return j;
186 }
187 if (currentSize == 0) {
188 LOGE("No sample size defined, cannot write data");
189 return j;
190 }
191 resize(currentSize);
192 }
193 }
194 return len;
195 }

251 void executeCallback(size_t size) {
252 size_t frameSize = size;
253 Frame frame;
254 frame.codec = codec;
255 frame.data = buffer.data();
256 frame.size = frameSize;
257 frame.timestamp = 0; // TODO: timestamp
258 switch (codec) {
259 case Codec::AAC: {
260 uint8_t out[frameSize + 7];
261 writeAdtsHeader(out, aacProfile, sampleRateIdx, channelCfg,
262 frameSize);
263 memcpy(out + 7, buffer.data(), frameSize);
264 frame.data = out;
265 frame.size = sizeof(out);
266 frame.mime = "audio/aac";
267 if (callback)
268 callback(frame, ref);
269 else
270 LOGE("No callback defined for audio frame extraction");
271 return;
272 }
273 case Codec::ALAC:
274 frame.mime = "audio/alac";
275 break;
276 case Codec::MP3:
277 frame.mime = "audio/mpeg";
278 break;
279 default:
280 frame.mime = nullptr;
281 break;
282 }
283 if (callback)
284 callback(frame, ref);
285 else
286 LOGE("No callback defined for audio frame extraction");
287 }

323 static void writeAdtsHeader(uint8_t* adts, int aacProfile,
324 int sampleRateIdx, int channelCfg,
325 int frameLen) {
326 adts[0] = 0xFF;
327 adts[1] = 0xF1;
328 adts[2] = ((aacProfile - 1) << 6) | (sampleRateIdx << 2) |
329 ((channelCfg >> 2) & 0x1);
330 adts[3] = ((channelCfg & 0x3) << 6) | ((frameLen + 7) >> 11);
331 adts[4] = ((frameLen + 7) >> 3) & 0xFF;
332 adts[5] = (((frameLen + 7) & 0x7) << 5) | 0x1F;
333 adts[6] = 0xFC;
334 }

459 static void boxDataSetupCallback(MP4Parser::Box& box, void* ref) {
460 M4AAudioDemuxer& self = *static_cast<M4AAudioDemuxer*>(ref);
461
462 // mdat must not be buffered
463 if (StrView(box.type) == "mdat") {
464 LOGI("Box: %s, size: %u bytes", box.type, (unsigned)box.size);
465 self.sampleExtractor.setMaxSize(box.size);
466 return;
467 }
468
469 bool is_relevant = isRelevantBox(box.type);
470 if (is_relevant) {
471 LOGI("Box: %s, size: %u bytes", box.type, (unsigned)box.size);
472 if (box.data_size == 0) {
473 // setup for incremental processing
474 self.resize(box.size);
475 self.buffer.clear();
476 } else {
477 // we have the complete box data
478 self.processBox(box);
479 }
480 }
481 }

491 static void incrementalBoxDataCallback(MP4Parser::Box& box,
492 const uint8_t* data, size_t len,
493 bool is_final, void* ref) {
494 M4AAudioDemuxer& self = *static_cast<M4AAudioDemuxer*>(ref);
495
496 // mdat must not be buffered
497 if (StrView(box.type) == "mdat") {
498 LOGI("*Box: %s, size: %u bytes", box.type, (unsigned)len);
499 self.sampleExtractor.write(data, len, is_final);
500 return;
501 }
502
503 // only process relevant boxes
504 if (!isRelevantBox(box.type)) return;
505
506 LOGI("*Box: %s, size: %u bytes", box.type, (unsigned)len);
507
508 // others fill buffer incrementally
509 if (len > 0) {
510 size_t written = self.buffer.writeArray(data, len);
511 if (written != len) {
512 LOGE("Failed to write all data to buffer, written: %zu, expected: %zu",
513 written, len);
514 }
515 }
516
517 // on last chunk, call the specific box handler
518 if (is_final) {
519 MP4Parser::Box complete_box = box;
520 complete_box.data = self.buffer.data();
521 complete_box.data_size = self.buffer.size();
522 self.processBox(complete_box);
523 // The buffer might be quite large, so we resize it to the default size
524 self.resize(self.default_size);
525 }

600 void fixALACMagicCookie(uint8_t* cookie, size_t len) {
601 if (len < 28) {
602 return;
603 }
604
605 // Helper to read/write big-endian
606 auto read32 = [](uint8_t* p) -> uint32_t {
607 return (p[0] << 24) | (p[1] << 16) | (p[2] << 8) | p[3];
608 };
609 auto write32 = [](uint8_t* p, uint32_t val) {
610 p[0] = (val >> 24) & 0xFF;
611 p[1] = (val >> 16) & 0xFF;
612 p[2] = (val >> 8) & 0xFF;
613 p[3] = val & 0xFF;
614 };
615 auto read16 = [](uint8_t* p) -> uint16_t { return (p[0] << 8) | p[1]; };
616 auto write16 = [](uint8_t* p, uint16_t val) {
617 p[0] = (val >> 8) & 0xFF;
618 p[1] = val & 0xFF;
619 };
620
621 // Fix values if zero or invalid
622 if (read32(cookie + 0) == 0) write32(cookie + 0, 4096); // frameLength
623 if (cookie[6] == 0) cookie[6] = 16; // bitDepth
624 if (cookie[7] == 0 || cookie[7] > 32) cookie[7] = 10; // pb
625 if (cookie[8] == 0 || cookie[8] > 32) cookie[8] = 14; // mb
626 if (cookie[9] == 0 || cookie[9] > 32) cookie[9] = 10; // kb
627 if (cookie[10] == 0 || cookie[10] > 8) cookie[10] = 2; // numChannels
628 if (read16(cookie + 11) == 0) write16(cookie + 11, 255); // maxRun
629 if (read32(cookie + 13) == 0) write32(cookie + 13, 8192); // maxFrameBytes
630 if (read32(cookie + 17) == 0) write32(cookie + 17, 512000); // avgBitRate
631 if (read32(cookie + 21) == 0) write32(cookie + 21, 44100); // sampleRate
632 }

656 void onStsz(MP4Parser::Box& box) {
657 LOGI("onStsz: %s, size: %zu bytes", box.type, box.data_size);
658 if (stsz_processed) return;
659 // Parse stsz box and fill sampleSizes
660 const uint8_t* data = box.data;
661 uint32_t sampleSize = readU32(data + 4);
662 uint32_t sampleCount = readU32(data + 8);
663 sampleExtractor.begin();
664 Vector<stsz_sample_size_t>& sampleSizes = sampleExtractor.getSampleSizes();
665 if (sampleSize == 0) {
666 LOGI("-> Sample Sizes Count: %u", sampleCount);
667 sampleSizes.resize(sampleCount);
668 for (uint32_t i = 0; i < sampleCount; ++i) {
669 uint32_t sampleSizes32 = readU32(data + 12 + i * 4);
670 sampleSizes[i] = static_cast<stsz_sample_size_t>(sampleSizes32);
671 assert(static_cast<uint32_t>(sampleSizes[i]) == sampleSizes32);
672 }
673 } else {
674 sampleExtractor.setFixedSampleCount(sampleSize, sampleCount);
675 }
676 stsz_processed = true;
677 }

683 void onStco(MP4Parser::Box& box) {
684 LOGI("onStco: %s, size: %zu bytes", box.type, box.data_size);
685 if (stco_processed) return;
686 // Parse stco box and fill chunkOffsets
687 const uint8_t* data = box.data + 4;
688 size_t size = box.data_size;
689 if (size < 4) return;
690 uint32_t entryCount = readU32(data);
691 Vector<uint32_t>& chunkOffsets = sampleExtractor.getChunkOffsets();
692 if (size < 4 + 4 * entryCount) return;
693 chunkOffsets.resize(entryCount);
694 LOGI("-> Chunk offsets count: %u", entryCount);
695 for (uint32_t i = 0; i < entryCount; ++i) {
696 chunkOffsets[i] = readU32(data + 4 + i * 4);
697 }
698 stco_processed = true;
699 }
