120 bool tryStartNewBox(size_t bufferSize) {
121 if (parseOffset + 8 > bufferSize) return false;
122 char type[5];
123
124 // get basic box information
125 parseOffset = checkParseOffset();
126 const uint8_t* p = buffer.data() + parseOffset;
127 uint32_t size32 = readU32(p);
128 strncpy(type, (char*)(p + 4), 4);
129 type[4] = '\0';
130 uint64_t boxSize = size32;
131 size_t headerSize = 8;
132
133 if (boxSize < headerSize) return false;
134
135 int level = static_cast<int>(levelStack.size());
136 bool is_container = isContainerBox(type);
137
138 if (is_container) {
139 handleContainerBox(type, boxSize, level);
140 return true;
141 }
142
143 size_t payload_size = static_cast<size_t>(boxSize - headerSize);
144 if (parseOffset + boxSize <= bufferSize) {
145 // start with full buffer!
146 handleCompleteBox(type, p, headerSize, payload_size, level);
147 parseOffset += boxSize;
148 } else {
149 startIncrementalBox(type, p, headerSize, payload_size, level, bufferSize);
150 return false; // Wait for more data
151 }
152 return true;
153 }

209 void startIncrementalBox(const char* type, const uint8_t* p,
210 size_t headerSize, size_t payload_size, int level,
211 size_t bufferSize) {
212 box_in_progress = true;
213 box_bytes_received = 0;
214 box_bytes_expected = payload_size;
215 strncpy(box_type, type, 5);
216 box_level = level;
217
218 size_t available_payload = bufferSize - parseOffset - headerSize;
219
220 if (available_payload > 0) {
221 box_bytes_received += available_payload;
222 if (incremental_data_callback) {
223 strcpy(box.type, box_type);
224 box.id = ++this->box.id;
225 box.data = nullptr;
226 box.data_size = available_payload;
227 box.level = box_level;
228 box.file_offset = 0;
229 box.is_complete = false;
230 box.is_container = false;
231
232 // regular callback from parent
233 box.size = payload_size;
234 box.data_size = 0;
235 processCallback(box);
236
237 // incremental callback
238 box.size = box_bytes_expected;
239 box.data_size = available_payload;
240 processIncrementalDataCallback(box, p + headerSize, available_payload, false, ref);
241 }
242 }
243 //fileOffset += (bufferSize - buffer.available());
244 fileOffset += (parseOffset + payload_size + 8);
245 buffer.clear();
246 parseOffset = 0;

253 bool continueIncrementalBox() {
254 size_t to_read = std::min((size_t)box_bytes_expected - box_bytes_received,
255 (size_t)buffer.available());
256 if (to_read == 0) return false;
257 if (incremental_data_callback) {
258 strcpy(box.type, box_type);
259 box.id = ++this->box.id;
260 box.data = nullptr;
261 box.size = box_bytes_expected;
262 box.data_size = to_read;
263 box.level = box_level;
264 box.file_offset = 0;
265 box.is_complete = (box_bytes_received + to_read == box_bytes_expected);
266 box.is_container = false;
267 processIncrementalDataCallback(box, buffer.data(), to_read, box.is_complete, ref);
268 }
269 box_bytes_received += to_read;
270 //fileOffset += to_read;
271 buffer.clearArray(to_read);
272
273 if (box_bytes_received >= box_bytes_expected) {
274 box_in_progress = false;
275 }
276 return true;
277 }

288 void processIncrementalDataCallback(Box& box, const uint8_t* data, size_t len,
289 bool is_final, void* ref) {
290 bool is_called = false;
291 bool call_generic = true;
292 for (auto& entry : incremental_data_callbacks) {
293 if (StrView(entry.type) == box.type) {
294 entry.cb(box, data, len, is_final, ref);
295 is_called = true;
296 if (!entry.callGeneric) call_generic = false;
297 break;
298 }
299 }
300 if ((!is_called || call_generic) && incremental_data_callback) {
301 incremental_data_callback(box, data, len, is_final, ref);
302 }
303 }
