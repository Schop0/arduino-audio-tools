19 size_t writeCodecConfig(const uint8_t* data, size_t len) override {
20 int32_t rc = dec.Init((void*)data, len);
21 if (rc != 0) {
22 LOGE("Init failed");
23 return 0;
24 }
25 AudioInfo info;
26 info.bits_per_sample = dec.mConfig.bitDepth;
27 info.channels = dec.mConfig.numChannels;
28 info.sample_rate = dec.mConfig.sampleRate;

30 is_init = true;
31 return len;
32 }

34 // define ALACSpecificConfig
35 size_t writeCodecConfig(ALACSpecificConfig config) {
36 return writeCodecConfig((uint8_t*)&config, sizeof(config));
37 }
38
39 bool begin(ALACSpecificConfig config) {
40 size_t written = writeCodecConfig(config);
41 return written == sizeof(ALACSpecificConfig);
42 }
43
45 size_t write(const uint8_t* encodedFrame, size_t frameLength) override {
46 LOGI("write: %d", (int)frameLength);
47 if (!is_init) {
48 ALACSpecificConfig config = {};
49 AudioInfo info = audioInfo();
50 config.frameLength = frameLength;
51 config.bitDepth = info.bits_per_sample;
52 config.numChannels = info.channels;
53 config.sampleRate = info.sample_rate;
54 writeCodecConfig((uint8_t*)&config, sizeof(config));
55 is_init = true;
56 }
57 // Make sure we have the output buffer set up
58 if (result_buffer.size() != outputBufferSize()) {
59 result_buffer.resize(outputBufferSize());
60 }
61
62 // Init bit buffer
63 struct BitBuffer bits;
64 BitBufferInit(&bits, (uint8_t*)encodedFrame, frameLength);
65
66 // Decode
67 uint32_t outNumSamples = 0;
68 int32_t status =
69 dec.Decode(&bits, result_buffer.data(), dec.mConfig.frameLength,
70 dec.mConfig.numChannels, &outNumSamples);
71
72 if (status != 0) {
73 LOGE("Decode failed with error: %d", status);
74 return 0;
75 }
76
77 // Process result
78 size_t outputSize =
79 outNumSamples * dec.mConfig.numChannels * dec.mConfig.bitDepth / 8;
80 size_t written = p_print->write(result_buffer.data(), outputSize);
81 if (outputSize != written) {
82 LOGE("write error: %d -> %d", outputSize, written);
83 }
84 return frameLength;
85 }
86
87 operator bool() { return true; }
88
89 protected:
90 ALACDecoder dec;
91 Vector<uint8_t> result_buffer;
92 bool is_init = false;
93
94 int outputBufferSize() {
95 return dec.mConfig.frameLength * dec.mConfig.numChannels *
96 dec.mConfig.bitDepth / 8;
97 }

106class EncoderALAC : public AudioEncoder {
107 public:
108 EncoderALAC(int bytesPerPacket = 1024) {
109 setDefaultBytesPerPacket(bytesPerPacket);
110 }
111 void setOutput(Print& out_stream) override { p_print = &out_stream; };
112
113 bool begin() override {
114 if (p_print == nullptr) {
115 LOGE("No output stream set");
116 return false;
117 }
118 // define input format
119 input_format.mSampleRate = info.sample_rate;
120 input_format.mFormatID = kALACFormatLinearPCM;
121 input_format.mFormatFlags = kALACFormatFlagIsSignedInteger;
122 input_format.mBytesPerPacket = default_bytes_per_packet;
123 input_format.mBytesPerFrame = info.channels * info.bits_per_sample / 8;
124 input_format.mFramesPerPacket =
125 default_bytes_per_packet / input_format.mBytesPerFrame;
126 input_format.mChannelsPerFrame = info.channels;
127 input_format.mBitsPerChannel = info.bits_per_sample;
128
129 // define output format
130 out_format = input_format;
131 out_format.mFormatID = kALACFormatAppleLossless;
132
133 int rc = enc.InitializeEncoder(out_format);
134
135 in_buffer.resize(default_bytes_per_packet);
136 out_buffer.resize(default_bytes_per_packet);
137 is_started = rc == 0;
138 return is_started;
139 }
140
141 void end() override {
142 enc.Finish();
143 is_started = false;
144 }
145
147 size_t write(const uint8_t* data, size_t len) override {
148 LOGI("write: %d", (int)len);
149 int32_t ioNumBytes = len;
150 for (int j = 0; j < len; j++) {
151 in_buffer.write(data[j]);
152 if (in_buffer.isFull()) {
153 int rc = enc.Encode(input_format, out_format, (uint8_t*)data,
154 out_buffer.data(), &ioNumBytes);
155 size_t written = p_print->write(out_buffer.data(), ioNumBytes);
156 if (ioNumBytes != written) {
157 LOGE("write error: %d -> %d", ioNumBytes, written);
158 }
159 in_buffer.reset();
160 }
161 }
162 return len;
163 }
164
165 ALACSpecificConfig config() {
166 enc.GetConfig(cfg);
167 return cfg;
168 }
169
170 operator bool() { return is_started && p_print != nullptr; }
171
172 const char* mime() override { return "audio/alac"; }
173
174 void setDefaultBytesPerPacket(int bytes) { default_bytes_per_packet = bytes; }
175
176 protected:
177 int default_bytes_per_packet = 1024;
178 ALACEncoder enc;
179 SingleBuffer<uint8_t> in_buffer;
180 Vector<uint8_t> out_buffer;
181 AudioFormatDescription input_format;
182 AudioFormatDescription out_format;
183 ALACSpecificConfig cfg;
184 Print* p_print = nullptr;
185 bool is_started = false;
186};
