46 // Parses esds content to extract audioObjectType, frequencyIndex, and
47 // channelConfiguration
48 bool parse(const uint8_t* data, size_t size) {
49 const uint8_t* ptr = data;
50 const uint8_t* end = data + size;
51
52 if (ptr + 4 > end) return false;
53 ptr += 4; // skip version + flags
54
55 if (ptr >= end || *ptr++ != 0x03) return false;
56 size_t es_len = parse_descriptor_length(ptr, end);
57 if (ptr + es_len > end) return false;
58
59 ptr += 2; // skip ES_ID
60 ptr += 1; // skip flags
61
62 if (ptr >= end || *ptr++ != 0x04) return false;
63 size_t dec_len = parse_descriptor_length(ptr, end);
64 if (ptr + dec_len > end) return false;
65
66 ptr += 13; // skip objectTypeIndication, streamType, bufferSizeDB,
67 // maxBitrate, avgBitrate
68
69 if (ptr >= end || *ptr++ != 0x05) return false;
70 size_t dsi_len = parse_descriptor_length(ptr, end);
71 if (ptr + dsi_len > end || dsi_len < 2) return false;
72
73 uint8_t byte1 = ptr[0];
74 uint8_t byte2 = ptr[1];
75
76 audioObjectType = (byte1 >> 3) & 0x1F;
77 samplingRateIndex = ((byte1 & 0x07) << 1) | ((byte2 >> 7) & 0x01);
78 channelConfiguration = (byte2 >> 3) & 0x0F;
79 return true;
80 }
81
82 protected:
83 // Helper to decode variable-length descriptor lengths (e.g. 0x80 80 80 05)
84 inline size_t parse_descriptor_length(const uint8_t*& ptr,
85 const uint8_t* end) {
86 size_t len = 0;
87 for (int i = 0; i < 4 && ptr < end; ++i) {
88 uint8_t b = *ptr++;
89 len = (len << 7) | (b & 0x7F);
90 if ((b & 0x80) == 0) break;
91 }
92 return len;
93 }
94 };

162 size_t write(const uint8_t* data, size_t len, bool is_final) {
163 // Resize buffer to the current sample size
164 size_t currentSize = currentSampleSize();
165 if (currentSize == 0) {
166 LOGE("No sample size defined: e.g. mdat before stsz!");
167 return 0;
168 }
169 resize(currentSize);
170
172 for (int j = 0; j < len; j++) {
173 buffer.write(data[j]);
174 if (buffer.available() == currentSize) {
175 LOGI("Sample# %zu: size %zu bytes", sampleIndex, currentSize);
176 executeCallback(currentSize);
177 buffer.clear();
178 box_pos += currentSize;
179 ++sampleIndex;
180 currentSize = currentSampleSize();
181 if (box_pos >= box_size) {
182 LOGI("Reached end of box: %s write",
183 is_final ? "final" : "not final");
184 return j;
185 }
186 if (currentSize == 0) {
187 LOGE("No sample size defined, cannot write data");
188 return j;
189 }
190 resize(currentSize);
191 }
192 }
193 return len;
194 }

250 void executeCallback(size_t size) {
251 size_t frameSize = size;
252 Frame frame;
253 frame.codec = codec;
254 frame.data = buffer.data();
255 frame.size = frameSize;
256 switch (codec) {
257 case Codec::AAC: {
258 uint8_t out[frameSize + 7];
259 writeAdtsHeader(out, aacProfile, sampleRateIdx, channelCfg,
260 frameSize);
261 memcpy(out + 7, buffer.data(), frameSize);
262 frame.data = out;
263 frame.size = sizeof(out);
264 frame.mime = "audio/aac";
265 if (callback)
266 callback(frame, ref);
267 else
268 LOGE("No callback defined for audio frame extraction");
269 return;
270 }
271 case Codec::ALAC:
272 frame.mime = "audio/alac";
273 break;
274 case Codec::MP3:
275 frame.mime = "audio/mpeg";
276 break;
277 default:
278 frame.mime = nullptr;
279 break;
280 }
281 if (callback)
282 callback(frame, ref);
283 else
284 LOGE("No callback defined for audio frame extraction");
285 }

321 static void writeAdtsHeader(uint8_t* adts, int aacProfile,
322 int sampleRateIdx, int channelCfg,
323 int frameLen) {
324 adts[0] = 0xFF;
325 adts[1] = 0xF1;
326 adts[2] = ((aacProfile - 1) << 6) | (sampleRateIdx << 2) |
327 ((channelCfg >> 2) & 0x1);
328 adts[3] = ((channelCfg & 0x3) << 6) | ((frameLen + 7) >> 11);
329 adts[4] = ((frameLen + 7) >> 3) & 0xFF;
330 adts[5] = (((frameLen + 7) & 0x7) << 5) | 0x1F;
331 adts[6] = 0xFC;
332 }

457 static void boxDataSetupCallback(MP4Parser::Box& box, void* ref) {
458 M4AAudioDemuxer& self = *static_cast<M4AAudioDemuxer*>(ref);
459
460 // mdat must not be buffered
461 if (StrView(box.type) == "mdat") {
462 LOGI("Box: %s, size: %u bytes", box.type, (unsigned)box.size);
463 self.sampleExtractor.setMaxSize(box.size);
464 return;
465 }
466
467 bool is_relevant = isRelevantBox(box.type);
468 if (is_relevant) {
469 LOGI("Box: %s, size: %u bytes", box.type, (unsigned)box.size);
470 if (box.data_size == 0) {
471 // setup for incremental processing
472 self.resize(box.size);
473 self.buffer.clear();
474 } else {
475 // we have the complete box data
476 self.processBox(box);
477 }
478 }
479 }

489 static void incrementalBoxDataCallback(MP4Parser::Box& box,
490 const uint8_t* data, size_t len,
491 bool is_final, void* ref) {
492 M4AAudioDemuxer& self = *static_cast<M4AAudioDemuxer*>(ref);
493
494 // mdat must not be buffered
495 if (StrView(box.type) == "mdat") {
496 LOGI("*Box: %s, size: %u bytes", box.type, (unsigned)len);
497 self.sampleExtractor.write(data, len, is_final);
498 return;
499 }
500
501 // only process relevant boxes
502 if (!isRelevantBox(box.type)) return;
503
504 LOGI("*Box: %s, size: %u bytes", box.type, (unsigned)len);
505
506 // others fill buffer incrementally
507 if (len > 0) {
508 size_t written = self.buffer.writeArray(data, len);
509 if (written != len) {
510 LOGE("Failed to write all data to buffer, written: %zu, expected: %zu",
511 written, len);
512 }
513 }
514
515 // on last chunk, call the specific box handler
516 if (is_final) {
517 MP4Parser::Box complete_box = box;
518 complete_box.data = self.buffer.data();
519 complete_box.data_size = self.buffer.size();
520 self.processBox(complete_box);
521 // The buffer might be quite large, so we resize it to the default size
522 self.resize(self.default_size);
523 }

598 void fixALACMagicCookie(uint8_t* cookie, size_t len) {
599 if (len < 28) {
600 return;
601 }
602
603 // Helper to read/write big-endian
604 auto read32 = [](uint8_t* p) -> uint32_t {
605 return (p[0] << 24) | (p[1] << 16) | (p[2] << 8) | p[3];
606 };
607 auto write32 = [](uint8_t* p, uint32_t val) {
608 p[0] = (val >> 24) & 0xFF;
609 p[1] = (val >> 16) & 0xFF;
610 p[2] = (val >> 8) & 0xFF;
611 p[3] = val & 0xFF;
612 };
613 auto read16 = [](uint8_t* p) -> uint16_t { return (p[0] << 8) | p[1]; };
614 auto write16 = [](uint8_t* p, uint16_t val) {
615 p[0] = (val >> 8) & 0xFF;
616 p[1] = val & 0xFF;
617 };
618
619 // Fix values if zero or invalid
620 if (read32(cookie + 0) == 0) write32(cookie + 0, 4096); // frameLength
621 if (cookie[6] == 0) cookie[6] = 16; // bitDepth
622 if (cookie[7] == 0 || cookie[7] > 32) cookie[7] = 10; // pb
623 if (cookie[8] == 0 || cookie[8] > 32) cookie[8] = 14; // mb
624 if (cookie[9] == 0 || cookie[9] > 32) cookie[9] = 10; // kb
625 if (cookie[10] == 0 || cookie[10] > 8) cookie[10] = 2; // numChannels
626 if (read16(cookie + 11) == 0) write16(cookie + 11, 255); // maxRun
627 if (read32(cookie + 13) == 0) write32(cookie + 13, 8192); // maxFrameBytes
628 if (read32(cookie + 17) == 0) write32(cookie + 17, 512000); // avgBitRate
629 if (read32(cookie + 21) == 0) write32(cookie + 21, 44100); // sampleRate
630 }

654 void onStsz(MP4Parser::Box& box) {
655 LOGI("onStsz: %s, size: %zu bytes", box.type, box.data_size);
656 if (stsz_processed) return;
657 // Parse stsz box and fill sampleSizes
658 const uint8_t* data = box.data;
659 uint32_t sampleSize = readU32(data + 4);
660 uint32_t sampleCount = readU32(data + 8);
661 sampleExtractor.begin();
662 Vector<stsz_sample_size_t>& sampleSizes = sampleExtractor.getSampleSizes();
663 if (sampleSize == 0) {
664 LOGI("-> Sample Sizes Count: %u", sampleCount);
665 sampleSizes.resize(sampleCount);
666 for (uint32_t i = 0; i < sampleCount; ++i) {
667 uint32_t sampleSizes32 = readU32(data + 12 + i * 4);
668 sampleSizes[i] = static_cast<stsz_sample_size_t>(sampleSizes32);
669 assert(static_cast<uint32_t>(sampleSizes[i]) == sampleSizes32);
670 }
671 } else {
672 sampleExtractor.setFixedSampleCount(sampleSize, sampleCount);
673 }
674 stsz_processed = true;
675 }

681 void onStco(MP4Parser::Box& box) {
682 LOGI("onStco: %s, size: %zu bytes", box.type, box.data_size);
683 if (stco_processed) return;
684 // Parse stco box and fill chunkOffsets
685 const uint8_t* data = box.data + 4;
686 size_t size = box.data_size;
687 if (size < 4) return;
688 uint32_t entryCount = readU32(data);
689 Vector<uint32_t>& chunkOffsets = sampleExtractor.getChunkOffsets();
690 if (size < 4 + 4 * entryCount) return;
691 chunkOffsets.resize(entryCount);
692 LOGI("-> Chunk offsets count: %u", entryCount);
693 for (uint32_t i = 0; i < entryCount; ++i) {
694 chunkOffsets[i] = readU32(data + 4 + i * 4);
695 }
696 stco_processed = true;
697 }
