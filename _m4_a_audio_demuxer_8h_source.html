46 // Parses esds content to extract audioObjectType, frequencyIndex, and
47 // channelConfiguration
48 bool parse(const uint8_t* data, size_t size) {
49 const uint8_t* ptr = data;
50 const uint8_t* end = data + size;
51
52 if (ptr + 4 > end) return false;
53 ptr += 4; // skip version + flags
54
55 if (ptr >= end || *ptr++ != 0x03) return false;
56 size_t es_len = parse_descriptor_length(ptr, end);
57 if (ptr + es_len > end) return false;
58
59 ptr += 2; // skip ES_ID
60 ptr += 1; // skip flags
61
62 if (ptr >= end || *ptr++ != 0x04) return false;
63 size_t dec_len = parse_descriptor_length(ptr, end);
64 if (ptr + dec_len > end) return false;
65
66 ptr += 13; // skip objectTypeIndication, streamType, bufferSizeDB,
67 // maxBitrate, avgBitrate
68
69 if (ptr >= end || *ptr++ != 0x05) return false;
70 size_t dsi_len = parse_descriptor_length(ptr, end);
71 if (ptr + dsi_len > end || dsi_len < 2) return false;
72
73 uint8_t byte1 = ptr[0];
74 uint8_t byte2 = ptr[1];
75
76 audioObjectType = (byte1 >> 3) & 0x1F;
77 samplingRateIndex = ((byte1 & 0x07) << 1) | ((byte2 >> 7) & 0x01);
78 channelConfiguration = (byte2 >> 3) & 0x0F;
79 return true;
80 }
81
82 protected:
83 // Helper to decode variable-length descriptor lengths (e.g. 0x80 80 80 05)
84 inline size_t parse_descriptor_length(const uint8_t*& ptr,
85 const uint8_t* end) {
86 size_t len = 0;
87 for (int i = 0; i < 4 && ptr < end; ++i) {
88 uint8_t b = *ptr++;
89 len = (len << 7) | (b & 0x7F);
90 if ((b & 0x80) == 0) break;
91 }
92 return len;
93 }
94 };

162 size_t write(const uint8_t* data, size_t len, bool is_final) {
163 // Resize buffer to the current sample size
164 size_t currentSize = currentSampleSize();
165 if (currentSize == 0) {
166 LOGE("No sample size defined: e.g. mdat before stsz!");
167 return 0;
168 }
169 resize(currentSize);
170
172 for (int j = 0; j < len; j++) {
173 buffer.write(data[j]);
174 if (buffer.available() == currentSize) {
175 LOGI("Sample# %zu: size %zu bytes", sampleIndex, currentSize);
176 executeCallback(currentSize);
177 buffer.clear();
178 box_pos += currentSize;
179 ++sampleIndex;
180 currentSize = currentSampleSize();
181 if (box_pos >= box_size) {
182 LOGI("Reached end of box: %s write",
183 is_final ? "final" : "not final");
184 return j;
185 }
186 if (currentSize == 0) {
187 LOGE("No sample size defined, cannot write data");
188 return j;
189 }
190 resize(currentSize);
191 }
192 }
193 return len;
194 }

237 Frame getFrame(size_t size, SingleBuffer<uint8_t>& buffer) {
238 Frame frame;
239 frame.codec = codec;
240 frame.data = buffer.data();
241 frame.size = size;
242 switch (codec) {
243 case Codec::AAC: {
244 // Prepare ADTS header + AAC frame
245 tmp.resize(size + 7);
246 writeAdtsHeader(tmp.data(), aacProfile, sampleRateIdx, channelCfg, size);
247 memcpy(tmp.data() + 7, buffer.data(), size);
248 frame.data = tmp.data();
249 frame.size = size + 7;
250 frame.mime = "audio/aac";
251 break;
252 }
253 case Codec::ALAC:
254 frame.mime = "audio/alac";
255 break;
256 case Codec::MP3:
257 frame.mime = "audio/mpeg";
258 break;
259 default:
260 frame.mime = nullptr;
261 break;
262 }
263 return frame;
264 }

328 static void writeAdtsHeader(uint8_t* adts, int aacProfile,
329 int sampleRateIdx, int channelCfg,
330 int frameLen) {
331 adts[0] = 0xFF;
332 adts[1] = 0xF1;
333 adts[2] = ((aacProfile - 1) << 6) | (sampleRateIdx << 2) |
334 ((channelCfg >> 2) & 0x1);
335 adts[3] = ((channelCfg & 0x3) << 6) | ((frameLen + 7) >> 11);
336 adts[4] = ((frameLen + 7) >> 3) & 0xFF;
337 adts[5] = (((frameLen + 7) & 0x7) << 5) | 0x1F;
338 adts[6] = 0xFC;
339 }

479 static void boxDataSetupCallback(MP4Parser::Box& box, void* ref) {
480 M4AAudioDemuxer& self = *static_cast<M4AAudioDemuxer*>(ref);
481
482 bool is_relevant = isRelevantBox(box.type);
483 if (is_relevant) {
484 LOGI("Box: %s, size: %u bytes", box.type, (unsigned)box.size);
485 if (box.data_size == 0) {
486 // setup for incremental processing
487 self.resize(box.size);
488 self.buffer.clear();
489 } else {
490 // we have the complete box data
491 self.processBox(box);
492 }
493 }

504 static void incrementalBoxDataCallback(MP4Parser::Box& box,
505 const uint8_t* data, size_t len,
506 bool is_final, void* ref) {
507 M4AAudioDemuxer& self = *static_cast<M4AAudioDemuxer*>(ref);
508
509 // only process relevant boxes
510 if (!isRelevantBox(box.type)) return;
511
512 LOGI("*Box: %s, size: %u bytes", box.type, (unsigned)len);
513
514 // others fill buffer incrementally
515 if (len > 0) {
516 size_t written = self.buffer.writeArray(data, len);
517 if (written != len) {
518 LOGE("Failed to write all data to buffer, written: %zu, expected: %zu",
519 written, len);
520 }
521 }
522
523 // on last chunk, call the specific box handler
524 if (is_final) {
525 MP4Parser::Box complete_box = box;
526 complete_box.data = self.buffer.data();
527 complete_box.data_size = self.buffer.size();
528 self.processBox(complete_box);
529 // The buffer might be quite large, so we resize it to the default size
530 self.resize(self.default_size);
531 }

606 void fixALACMagicCookie(uint8_t* cookie, size_t len) {
607 if (len < 28) {
608 return;
609 }
610
611 // Helper to read/write big-endian
612 auto read32 = [](uint8_t* p) -> uint32_t {
613 return (p[0] << 24) | (p[1] << 16) | (p[2] << 8) | p[3];
614 };
615 auto write32 = [](uint8_t* p, uint32_t val) {
616 p[0] = (val >> 24) & 0xFF;
617 p[1] = (val >> 16) & 0xFF;
618 p[2] = (val >> 8) & 0xFF;
619 p[3] = val & 0xFF;
620 };
621 auto read16 = [](uint8_t* p) -> uint16_t { return (p[0] << 8) | p[1]; };
622 auto write16 = [](uint8_t* p, uint16_t val) {
623 p[0] = (val >> 8) & 0xFF;
624 p[1] = val & 0xFF;
625 };
626
627 // Fix values if zero or invalid
628 if (read32(cookie + 0) == 0) write32(cookie + 0, 4096); // frameLength
629 if (cookie[6] == 0) cookie[6] = 16; // bitDepth
630 if (cookie[7] == 0 || cookie[7] > 32) cookie[7] = 10; // pb
631 if (cookie[8] == 0 || cookie[8] > 32) cookie[8] = 14; // mb
632 if (cookie[9] == 0 || cookie[9] > 32) cookie[9] = 10; // kb
633 if (cookie[10] == 0 || cookie[10] > 8) cookie[10] = 2; // numChannels
634 if (read16(cookie + 11) == 0) write16(cookie + 11, 255); // maxRun
635 if (read32(cookie + 13) == 0) write32(cookie + 13, 8192); // maxFrameBytes
636 if (read32(cookie + 17) == 0) write32(cookie + 17, 512000); // avgBitRate
637 if (read32(cookie + 21) == 0) write32(cookie + 21, 44100); // sampleRate
638 }

662 void onStsz(MP4Parser::Box& box) {
663 LOGI("onStsz: %s, size: %zu bytes", box.type, box.data_size);
664 if (stsz_processed) return;
665 // Parse stsz box and fill sampleSizes
666 const uint8_t* data = box.data;
667 uint32_t sampleSize = readU32(data + 4);
668 uint32_t sampleCount = readU32(data + 8);
669 sampleExtractor.begin();
670 Vector<stsz_sample_size_t>& sampleSizes = sampleExtractor.getSampleSizes();
671 if (sampleSize == 0) {
672 LOGI("-> Sample Sizes Count: %u", sampleCount);
673 sampleSizes.resize(sampleCount);
674 for (uint32_t i = 0; i < sampleCount; ++i) {
675 uint32_t sampleSizes32 = readU32(data + 12 + i * 4);
676 sampleSizes[i] = static_cast<stsz_sample_size_t>(sampleSizes32);
677 assert(static_cast<uint32_t>(sampleSizes[i]) == sampleSizes32);
678 }
679 } else {
680 sampleExtractor.setFixedSampleCount(sampleSize, sampleCount);
681 }
682 stsz_processed = true;
683 }

689 void onStco(MP4Parser::Box& box) {
690 LOGI("onStco: %s, size: %zu bytes", box.type, box.data_size);
691 if (stco_processed) return;
692 // Parse stco box and fill chunkOffsets
693 const uint8_t* data = box.data + 4;
694 size_t size = box.data_size;
695 if (size < 4) return;
696 uint32_t entryCount = readU32(data);
697 Vector<uint32_t>& chunkOffsets = sampleExtractor.getChunkOffsets();
698 if (size < 4 + 4 * entryCount) return;
699 chunkOffsets.resize(entryCount);
700 LOGI("-> Chunk offsets count: %u", entryCount);
701 for (uint32_t i = 0; i < entryCount; ++i) {
702 chunkOffsets[i] = readU32(data + 4 + i * 4);
703 }
704 stco_processed = true;
705 }

707 void printHexDump(const MP4Parser::Box& box) {
708 const uint8_t* data = box.data;
709 size_t len = box.data_size;
710 LOGI("===========================");
711 for (size_t i = 0; i < len; i += 16) {
712 char hex[49] = {0};
713 char ascii[17] = {0};
714 for (size_t j = 0; j < 16 && i + j < len; ++j) {
715 sprintf(hex + j * 3, "%02X ", data[i + j]);
716 ascii[j] = (data[i + j] >= 32 && data[i + j] < 127) ? data[i + j] : '.';
717 }
718 ascii[16] = 0;
719 LOGI("%04zx: %-48s |%s|", i, hex, ascii);
720 }
721 LOGI("===========================");
