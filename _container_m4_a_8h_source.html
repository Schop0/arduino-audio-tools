102 static void decodeAudio(const M4AAudioDemuxer::Frame& frame, void* ref) {
103 ContainerM4A* self = static_cast<ContainerM4A*>(ref);
104 if (self->p_decoder == nullptr) {
105 LOGE("No decoder defined, cannot decode audio frame: %s (%u bytes)", frame.mime, (unsigned) frame.size);
106 return;
107 }
108 MultiDecoder& dec = *(self->p_decoder);
109 // select decoder based on mime type
110 if (!dec.selectDecoder(frame.mime)) {
111 const char* mime = frame.mime ? frame.mime : "(nullptr)";
112 LOGE("No decoder found for mime type: %s", mime);
113 return;
114 }
115
116 // for ALAC only: process magic cookie if not done yet
117 if (StrView(frame.mime) == "audio/alac" &&
118 !self->is_magic_cookie_processed) {
119 auto& magic_cookie = self->demux.getALACMagicCookie();
120 if (magic_cookie.size() > 0) {
121 if (!dec.setCodecConfig(magic_cookie.data(), magic_cookie.size())){
122 LOGE("Failed to set ALAC magic cookie for decoder: %s", dec.selectedMime());
123 }
124 }
125 self->is_magic_cookie_processed = true;
126 }
127 // write encoded data to decoder
128 dec.write(frame.data, frame.size);
129 }
