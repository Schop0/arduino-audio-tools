44 // Parses esds content to extract audioObjectType, frequencyIndex, and
45 // channelConfiguration
46 bool parse(const uint8_t* data, size_t size) {
47 const uint8_t* ptr = data;
48 const uint8_t* end = data + size;
49
50 if (ptr + 4 > end) return false;
51 ptr += 4; // skip version + flags
52
53 if (ptr >= end || *ptr++ != 0x03) return false;
54 size_t es_len = parse_descriptor_length(ptr, end);
55 if (ptr + es_len > end) return false;
56
57 ptr += 2; // skip ES_ID
58 ptr += 1; // skip flags
59
60 if (ptr >= end || *ptr++ != 0x04) return false;
61 size_t dec_len = parse_descriptor_length(ptr, end);
62 if (ptr + dec_len > end) return false;
63
64 ptr += 13; // skip objectTypeIndication, streamType, bufferSizeDB,
65 // maxBitrate, avgBitrate
66
67 if (ptr >= end || *ptr++ != 0x05) return false;
68 size_t dsi_len = parse_descriptor_length(ptr, end);
69 if (ptr + dsi_len > end || dsi_len < 2) return false;
70
71 uint8_t byte1 = ptr[0];
72 uint8_t byte2 = ptr[1];
73
74 audioObjectType = (byte1 >> 3) & 0x1F;
75 samplingRateIndex = ((byte1 & 0x07) << 1) | ((byte2 >> 7) & 0x01);
76 channelConfiguration = (byte2 >> 3) & 0x0F;
77 return true;
78 }
79
80 protected:
81 // Helper to decode variable-length descriptor lengths (e.g. 0x80 80 80 05)
82 inline size_t parse_descriptor_length(const uint8_t*& ptr,
83 const uint8_t* end) {
84 size_t len = 0;
85 for (int i = 0; i < 4 && ptr < end; ++i) {
86 uint8_t b = *ptr++;
87 len = (len << 7) | (b & 0x7F);
88 if ((b & 0x80) == 0) break;
89 }
90 return len;
91 }
92 };

160 size_t write(const uint8_t* data, size_t len, bool is_final) {
161 // Resize buffer to the current sample size
162 size_t currentSize = currentSampleSize();
163 if (currentSize == 0) {
164 LOGE("No sample size defined: e.g. mdat before stsz!");
165 return 0;
166 }
167 resize(currentSize);
168
170 for (int j = 0; j < len; j++) {
171 buffer.write(data[j]);
172 if (buffer.available() == currentSize) {
173 LOGI("Sample# %zu: size %zu bytes", sampleIndex, currentSize);
174 executeCallback(currentSize);
175 buffer.clear();
176 box_pos += currentSize;
177 ++sampleIndex;
178 currentSize = currentSampleSize();
179 if (box_pos >= box_size) {
180 LOGI("Reached end of box: %s write",
181 is_final ? "final" : "not final");
182 return j;
183 }
184 if (currentSize == 0) {
185 LOGE("No sample size defined, cannot write data");
186 return j;
187 }
188 resize(currentSize);
189 }
190 }
191 return len;
192 }

235 Frame getFrame(size_t size, SingleBuffer<uint8_t>& buffer) {
236 Frame frame;
237 frame.codec = codec;
238 frame.data = buffer.data();
239 frame.size = size;
240 switch (codec) {
241 case Codec::AAC: {
242 // Prepare ADTS header + AAC frame
243 tmp.resize(size + 7);
244 writeAdtsHeader(tmp.data(), aacProfile, sampleRateIdx, channelCfg,
245 size);
246 memcpy(tmp.data() + 7, buffer.data(), size);
247 frame.data = tmp.data();
248 frame.size = size + 7;
249 frame.mime = "audio/aac";
250 break;
251 }
252 case Codec::ALAC:
253 frame.mime = "audio/alac";
254 break;
255 case Codec::MP3:
256 frame.mime = "audio/mpeg";
257 break;
258 default:
259 frame.mime = nullptr;
260 break;
261 }
262 return frame;
263 }

327 static void writeAdtsHeader(uint8_t* adts, int aacProfile,
328 int sampleRateIdx, int channelCfg,
329 int frameLen) {
330 adts[0] = 0xFF;
331 adts[1] = 0xF1;
332 adts[2] = ((aacProfile - 1) << 6) | (sampleRateIdx << 2) |
333 ((channelCfg >> 2) & 0x1);
334 adts[3] = ((channelCfg & 0x3) << 6) | ((frameLen + 7) >> 11);
335 adts[4] = ((frameLen + 7) >> 3) & 0xFF;
336 adts[5] = (((frameLen + 7) & 0x7) << 5) | 0x1F;
337 adts[6] = 0xFC;
338 }

446 void fixALACMagicCookie(uint8_t* cookie, size_t len) {
447 if (len < 28) {
448 return;
449 }
450
451 // Helper to read/write big-endian
452 auto read32 = [](uint8_t* p) -> uint32_t {
453 return (p[0] << 24) | (p[1] << 16) | (p[2] << 8) | p[3];
454 };
455 auto write32 = [](uint8_t* p, uint32_t val) {
456 p[0] = (val >> 24) & 0xFF;
457 p[1] = (val >> 16) & 0xFF;
458 p[2] = (val >> 8) & 0xFF;
459 p[3] = val & 0xFF;
460 };
461 auto read16 = [](uint8_t* p) -> uint16_t { return (p[0] << 8) | p[1]; };
462 auto write16 = [](uint8_t* p, uint16_t val) {
463 p[0] = (val >> 8) & 0xFF;
464 p[1] = val & 0xFF;
465 };
466
467 // Fix values if zero or invalid
468 if (read32(cookie + 0) == 0) write32(cookie + 0, 4096); // frameLength
469 if (cookie[6] == 0) cookie[6] = 16; // bitDepth
470 if (cookie[7] == 0 || cookie[7] > 32) cookie[7] = 10; // pb
471 if (cookie[8] == 0 || cookie[8] > 32) cookie[8] = 14; // mb
472 if (cookie[9] == 0 || cookie[9] > 32) cookie[9] = 10; // kb
473 if (cookie[10] == 0 || cookie[10] > 8) cookie[10] = 2; // numChannels
474 if (read16(cookie + 11) == 0) write16(cookie + 11, 255); // maxRun
475 if (read32(cookie + 13) == 0) write32(cookie + 13, 8192); // maxFrameBytes
476 if (read32(cookie + 17) == 0) write32(cookie + 17, 512000); // avgBitRate
477 if (read32(cookie + 21) == 0) write32(cookie + 21, 44100); // sampleRate
478 }

502 void onStsz(MP4Parser::Box& box) {
503 LOGI("onStsz: %s, size: %zu bytes", box.type, box.data_size);
504 if (stsz_processed) return;
505 // Parse stsz box and fill sampleSizes
506 const uint8_t* data = box.data;
507 uint32_t sampleSize = readU32(data + 4);
508 uint32_t sampleCount = readU32(data + 8);
509 sampleExtractor.begin();
510 Vector<stsz_sample_size_t>& sampleSizes = sampleExtractor.getSampleSizes();
511 if (sampleSize == 0) {
512 LOGI("-> Sample Sizes Count: %u", sampleCount);
513 sampleSizes.resize(sampleCount);
514 for (uint32_t i = 0; i < sampleCount; ++i) {
515 uint32_t sampleSizes32 = readU32(data + 12 + i * 4);
516 sampleSizes[i] = static_cast<stsz_sample_size_t>(sampleSizes32);
517 assert(static_cast<uint32_t>(sampleSizes[i]) == sampleSizes32);
518 }
519 } else {
520 sampleExtractor.setFixedSampleCount(sampleSize, sampleCount);
521 }
522 stsz_processed = true;
523 }

529 void onStco(MP4Parser::Box& box) {
530 LOGI("onStco: %s, size: %zu bytes", box.type, box.data_size);
531 if (stco_processed) return;
532 // Parse stco box and fill chunkOffsets
533 const uint8_t* data = box.data + 4;
534 size_t size = box.data_size;
535 if (size < 4) return;
536 uint32_t entryCount = readU32(data);
537 Vector<uint32_t>& chunkOffsets = sampleExtractor.getChunkOffsets();
538 if (size < 4 + 4 * entryCount) return;
539 chunkOffsets.resize(entryCount);
540 LOGI("-> Chunk offsets count: %u", entryCount);
541 for (uint32_t i = 0; i < entryCount; ++i) {
542 chunkOffsets[i] = readU32(data + 4 + i * 4);
543 }
544 stco_processed = true;
545 }

547 void printHexDump(const MP4Parser::Box& box) {
548 const uint8_t* data = box.data;
549 size_t len = box.data_size;
550 LOGI("===========================");
551 for (size_t i = 0; i < len; i += 16) {
552 char hex[49] = {0};
553 char ascii[17] = {0};
554 for (size_t j = 0; j < 16 && i + j < len; ++j) {
555 sprintf(hex + j * 3, "%02X ", data[i + j]);
556 ascii[j] = (data[i + j] >= 32 && data[i + j] < 127) ? data[i + j] : '.';
557 }
558 ascii[16] = 0;
559 LOGI("%04zx: %-48s |%s|", i, hex, ascii);
