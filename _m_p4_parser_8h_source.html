168 int parseString(const uint8_t* str, int len, int fileOffset = 0,
169 int level = 0) {
170 char type[5];
171 int idx = 0;
172 Box box;
173 while (true) {
174 if (!isValidType((const char*)str + idx + 4)) {
175 return idx;
176 }
177 size_t box_size = readU32(str + idx) - 8;
178 box.data = str + 8 + idx;
179 box.size = box_size;
180 box.level = level;
181 box.data_size = box.size;
182 box.file_offset = fileOffset + idx;
183 strncpy(box.type, (char*)(str + idx + 4), 4);
184 box.type[4] = '\0';
185 idx += box.size;
186 processCallback(box);
187 if (idx >= len) break; // No more data to parse
188 }
189 return idx;
190 }

193 bool findBox(const char* name, const uint8_t* data, size_t len, Box& result) {
194 for (int j = 0; j < len - 4; j++) {
195 if (!isValidType((const char*)data + j + 4)) {
196 continue; // Skip invalid types
197 }
198 size_t box_size = readU32(data + j) - 8;
199 if (box_size < 8) continue; // Invalid box size
200 Box box;
201 box.data = data + j + 8;
202 box.size = box_size;
203 box.data_size = box.size;
204 strncpy(box.type, (char*)(data + j + 4), 4);
205 box.type[4] = '\0';
206 if (StrView(box.type) == name) {
207 result = box;
208 return true; // Found the box
209 }
210 }
211 return false;
212 }

282 virtual void parse() {
283 while (true) {
284 size_t bufferSize = buffer.available();
285 // not enough data to parse a box header
286 if (parseOffset + 8 > bufferSize) break;
287
288 parseOffset = checkParseOffset();
289
290 const uint8_t* p = buffer.data() + parseOffset;
291 uint32_t size32 = readU32(p);
292 char type[5];
293 strncpy(type, (char*)(p + 4), 4);
294
295 uint64_t boxSize = size32;
296 size_t headerSize = 8;
297 bool is_container = isContainerBox(type);
298
299 // check data in buffer
300 if (!is_container) {
301 // not enough data to provide data
302 if (boxSize > buffer.size()) {
303 LOGE("MP4Parser: '%s' box size %u exceeds buffer size %u", type,
304 (unsigned)boxSize, (unsigned)buffer.size());
305 is_error = true;
306 return;
307 }
308 // not enough data in buffer
309 if (parseOffset + boxSize > bufferSize) break;
310 }
311
312 // fill box with data
313 // Calculate absolute file offset for this box
314 uint64_t absBoxOffset = fileOffset + parseOffset;
315 int level = levelStack.size();
316
317 strcpy(box.type, type);
318 box.id++;
319 box.data = p + headerSize;
320 box.size = static_cast<size_t>(boxSize - headerSize);
321 box.data_size = box.size;
322 box.level = level;
323 box.file_offset = fileOffset + parseOffset;
324 box.is_complete = (parseOffset + boxSize <= bufferSize);
325 box.is_container = is_container;
326
327 // Special logic for container: usually no data
328 if (box.is_container) {
329 box.data_size = getContainerDataLength(box.type);
330 if (box.data_size == 0) box.data = nullptr;
331 box.is_complete = true;
332 }
333
334 // Regular logic for box with complete data
335 processCallback(box);
336
337 // Recurse into container
338 if (box.is_container) {
339 levelStack.push_back(absBoxOffset + boxSize);
340 parseOffset += (headerSize + box.data_size);
341 continue;
342 }
343
344 parseOffset += boxSize;
345
346 popLevels();
347 }
348
349 if (parseOffset > 0) {
350 fileOffset += parseOffset;
351 buffer.clearArray(parseOffset);
352 parseOffset = 0;
353 }
354 }

359 void popLevels() {
360 // Pop levels if we've passed their bounds (absolute file offset)
361 while (!levelStack.empty() &&
362 (fileOffset + parseOffset) >= levelStack.back()) {
363 levelStack.pop_back();
364 }
365 }

373 void processCallback(Box& box) {
374 bool is_called = false;
375 bool call_generic = true;
376 for (const auto& entry : callbacks) {
377 if (strncmp(entry.type, box.type, 4) == 0) {
378 entry.cb(box, ref);
379 is_called = true;
380 if (!entry.callGeneric) call_generic = false;
381 }
382 }
384 if ((!is_called || call_generic) && callback) callback(box, ref);
385 }

392 bool isContainerBox(const char* type) {
393 // fill with default values if nothing has been defined
394 if (containers.empty()) {
395 // pure containers
396 static const char* containers_str[] = {
397 "moov", "trak", "mdia", "minf", "stbl", "edts", "dinf", "udta",

402 info.start = 0;
403 containers.push_back(info);

447 size_t checkParseOffset() {
448 size_t current = parseOffset;
449 const char* type = (char*)(buffer.data() + parseOffset + 4);
450 for (int j = 0; j < buffer.available() - parseOffset - 4; j += 4) {
451 if (isValidType(type, j)) {
452 if (j != 0) {
453 // report the data under the last valid box
454 box.size = 0;
455 box.data_size = j;
456 box.level = static_cast<int>(levelStack.size()) + 1;
457 box.data = buffer.data() + parseOffset;
458 processCallback(box);
459 }
460
461 return j + parseOffset;
462 }
463 }
464 return parseOffset;
