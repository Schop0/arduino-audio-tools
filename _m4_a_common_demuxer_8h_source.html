44 // Parses esds content to extract audioObjectType, frequencyIndex, and
45 // channelConfiguration
46 bool parse(const uint8_t* data, size_t size) {
47 const uint8_t* ptr = data;
48 const uint8_t* end = data + size;
49
50 if (ptr + 4 > end) return false;
51 ptr += 4; // skip version + flags
52
53 if (ptr >= end || *ptr++ != 0x03) return false;
54 size_t es_len = parse_descriptor_length(ptr, end);
55 if (ptr + es_len > end) return false;
56
57 ptr += 2; // skip ES_ID
58 ptr += 1; // skip flags
59
60 if (ptr >= end || *ptr++ != 0x04) return false;
61 size_t dec_len = parse_descriptor_length(ptr, end);
62 if (ptr + dec_len > end) return false;
63
64 ptr += 13; // skip objectTypeIndication, streamType, bufferSizeDB,
65 // maxBitrate, avgBitrate
66
67 if (ptr >= end || *ptr++ != 0x05) return false;
68 size_t dsi_len = parse_descriptor_length(ptr, end);
69 if (ptr + dsi_len > end || dsi_len < 2) return false;
70
71 uint8_t byte1 = ptr[0];
72 uint8_t byte2 = ptr[1];
73
74 audioObjectType = (byte1 >> 3) & 0x1F;
75 samplingRateIndex = ((byte1 & 0x07) << 1) | ((byte2 >> 7) & 0x01);
76 channelConfiguration = (byte2 >> 3) & 0x0F;
77 return true;
78 }
79
80 protected:
81 // Helper to decode variable-length descriptor lengths (e.g. 0x80 80 80 05)
82 inline size_t parse_descriptor_length(const uint8_t*& ptr,
83 const uint8_t* end) {
84 size_t len = 0;
85 for (int i = 0; i < 4 && ptr < end; ++i) {
86 uint8_t b = *ptr++;
87 len = (len << 7) | (b & 0x7F);
88 if ((b & 0x80) == 0) break;
89 }
90 return len;
91 }
92 };

161 size_t write(const uint8_t* data, size_t len, bool is_final) {
162 // Resize buffer to the current sample size
163 size_t currentSize = currentSampleSize();
164 if (currentSize == 0) {
165 LOGE("No sample size defined: e.g. mdat before stsz!");
166 return 0;
167 }
168 resize(currentSize);
169
171 for (int j = 0; j < len; j++) {
172 buffer.write(data[j]);
173 if (buffer.available() >= currentSize) {
174 LOGI("Sample# %zu: size %zu bytes", sampleIndex, currentSize);
175 executeCallback(currentSize);
176 buffer.clear();
177 box_pos += currentSize;
178 ++sampleIndex;
179 currentSize = currentSampleSize();
180 if (box_pos >= box_size) {
181 LOGI("Reached end of box: %s write",
182 is_final ? "final" : "not final");
183 return j;
184 }
185 if (currentSize == 0) {
186 LOGE("No sample size defined, cannot write data");
187 return j;
188 }
189 }
190 }
191 return len;
192 }

247 Frame getFrame(size_t size, SingleBuffer<uint8_t>& buffer) {
248 Frame frame;
249 frame.codec = codec;
250 frame.data = buffer.data();
251 frame.size = size;
252 switch (codec) {
253 case Codec::AAC: {
254 // Prepare ADTS header + AAC frame
255 tmp.resize(size + 7);
256 writeAdtsHeader(tmp.data(), aacProfile, sampleRateIdx, channelCfg,
257 size);
258 memcpy(tmp.data() + 7, buffer.data(), size);
259 frame.data = tmp.data();
260 frame.size = size + 7;
261 frame.mime = "audio/aac";
262 break;
263 }
264 case Codec::ALAC:
265 frame.mime = "audio/alac";
266 break;
267 case Codec::MP3:
268 frame.mime = "audio/mpeg";
269 break;
270 default:
271 frame.mime = nullptr;
272 break;
273 }
274 return frame;
275 }

352 static void writeAdtsHeader(uint8_t* adts, int aacProfile,
353 int sampleRateIdx, int channelCfg,
354 int frameLen) {
355 adts[0] = 0xFF;
356 adts[1] = 0xF1;
357 adts[2] = ((aacProfile - 1) << 6) | (sampleRateIdx << 2) |
358 ((channelCfg >> 2) & 0x1);
359 adts[3] = ((channelCfg & 0x3) << 6) | ((frameLen + 7) >> 11);
360 adts[4] = ((frameLen + 7) >> 3) & 0xFF;
361 adts[5] = (((frameLen + 7) & 0x7) << 5) | 0x1F;
362 adts[6] = 0xFC;
363 }

481 void fixALACMagicCookie(uint8_t* cookie, size_t len) {
482 if (len < 28) {
483 return;
484 }
485
486 // Helper to read/write big-endian
487 auto read32 = [](uint8_t* p) -> uint32_t {
488 return (p[0] << 24) | (p[1] << 16) | (p[2] << 8) | p[3];
489 };
490 auto write32 = [](uint8_t* p, uint32_t val) {
491 p[0] = (val >> 24) & 0xFF;
492 p[1] = (val >> 16) & 0xFF;
493 p[2] = (val >> 8) & 0xFF;
494 p[3] = val & 0xFF;
495 };
496 auto read16 = [](uint8_t* p) -> uint16_t { return (p[0] << 8) | p[1]; };
497 auto write16 = [](uint8_t* p, uint16_t val) {
498 p[0] = (val >> 8) & 0xFF;
499 p[1] = val & 0xFF;
500 };
501
502 // Fix values if zero or invalid
503 if (read32(cookie + 0) == 0) write32(cookie + 0, 4096); // frameLength
504 if (cookie[6] == 0) cookie[6] = 16; // bitDepth
505 if (cookie[7] == 0 || cookie[7] > 32) cookie[7] = 10; // pb
506 if (cookie[8] == 0 || cookie[8] > 32) cookie[8] = 14; // mb
507 if (cookie[9] == 0 || cookie[9] > 32) cookie[9] = 10; // kb
508 if (cookie[10] == 0 || cookie[10] > 8) cookie[10] = 2; // numChannels
509 if (read16(cookie + 11) == 0) write16(cookie + 11, 255); // maxRun
510 if (read32(cookie + 13) == 0) write32(cookie + 13, 8192); // maxFrameBytes
511 if (read32(cookie + 17) == 0) write32(cookie + 17, 512000); // avgBitRate
512 if (read32(cookie + 21) == 0) write32(cookie + 21, 44100); // sampleRate
513 }

537 void onStsz(MP4Parser::Box& box) {
538 LOGI("onStsz: %s, size: %zu bytes", box.type, box.data_size);
539 if (stsz_processed) return;
540 // Parse stsz box and fill sampleSizes
541 const uint8_t* data = box.data;
542 uint32_t sampleSize = readU32(data + 4);
543 uint32_t sampleCount = readU32(data + 8);
544 sampleExtractor.begin();
545 BaseBuffer<stsz_sample_size_t>& sampleSizes = sampleExtractor.getSampleSizesBuffer();
546 if (sampleSize == 0) {
547 LOGI("-> Sample Sizes Count: %u", sampleCount);
548 sampleSizes.resize(sampleCount);
549 for (uint32_t i = 0; i < sampleCount; ++i) {
550 uint32_t sampleSizes32 = readU32(data + 12 + i * 4);
551 // if this is giving an error change the stsz_sample_size_t
552 assert(sampleSizes32 <= UINT16_MAX);
553 stsz_sample_size_t sampleSizes16 = sampleSizes32;
554 assert(sampleSizes.write(sampleSizes16));
555 }
556 } else {
557 sampleExtractor.setFixedSampleCount(sampleSize, sampleCount);
558 }
559 stsz_processed = true;
560 }

566 void onStco(MP4Parser::Box& box) {
567 LOGI("onStco: %s, size: %zu bytes", box.type, box.data_size);
568 if (stco_processed) return;
569 // Parse stco box and fill chunkOffsets
570 const uint8_t* data = box.data + 4;
571 size_t size = box.data_size;
572 if (size < 4) return;
573 uint32_t entryCount = readU32(data);
574 BaseBuffer<uint32_t>& chunkOffsets = sampleExtractor.getChunkOffsetsBuffer();
575 if (size < 4 + 4 * entryCount) return;
576 chunkOffsets.resize(entryCount);
577 LOGI("-> Chunk offsets count: %u", entryCount);
578 for (uint32_t i = 0; i < entryCount; ++i) {
579 chunkOffsets.write(readU32(data + 4 + i * 4));
580 }
581 stco_processed = true;
582 }

584 void printHexDump(const MP4Parser::Box& box) {
585 const uint8_t* data = box.data;
586 size_t len = box.data_size;
587 LOGI("===========================");
588 for (size_t i = 0; i < len; i += 16) {
589 char hex[49] = {0};
590 char ascii[17] = {0};
591 for (size_t j = 0; j < 16 && i + j < len; ++j) {
592 sprintf(hex + j * 3, "%02X ", data[i + j]);
593 ascii[j] = (data[i + j] >= 32 && data[i + j] < 127) ? data[i + j] : '.';
594 }
595 ascii[16] = 0;
596 LOGI("%04zx: %-48s |%s|", i, hex, ascii);
597 }
598 LOGI("===========================");
